#alias 64: TYPE_SHEEP


sub ObjectMain
	Object.YPos+=Object.Value0
	Object.Value0+=10240
	switch Object.State
	case 0
		Object.XPos+=65536
		if Object.Value0>0
			ObjectTileCollision(0,0,8,0)
			if CheckResult==1
				Object.Direction=FACING_LEFT
				Object.Value0=-327680
				Object.State=1
			endif
		endif
		break
	case 1
		Object.XPos-=65536
		if Object.Value0>0
			ObjectTileCollision(0,0,8,0)
			if CheckResult==1
				Object.Direction=FACING_RIGHT
				Object.Value0=-327680
				Object.State=0
			endif
		endif
		break
	endswitch
endsub


sub ObjectDraw
	if MetalSonic_Destroyed==1
		if Object.Value0<0
			DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
		else
			DrawSpriteFX(1,FX_FLIP,Object.XPos,Object.YPos)
		endif
	endif
	if Object[24].PropertyValue==2
		if Object.Value0<0
			DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
		else
			DrawSpriteFX(1,FX_FLIP,Object.XPos,Object.YPos)
		endif
	endif
endsub

	LoadSpriteSheet("R8/Objects2.gif")

		if Object[ArrayPos0].Type==TypeName[Sheep]

		endif

endsub
