sub ObjectMain
	switch Object.State
	case 1
		if Object.Value0<55
			Object.Value0++
			Object.Value1-=131072
		else
			Object.State=2
		endif
		break
	endswitch
endsub


sub ObjectPlayerInteraction
	if Object.State==0
		TempValue0=Player.CollisionRight
		TempValue0<<=16
		TempValue0+=Player.XPos
		TempValue1=Object.Value1
		if TempValue0>TempValue1
			Player.XVelocity=0
			Player.Speed=0
			Player.XPos=TempValue1
			TempValue0=Player.CollisionRight
			TempValue0<<=16
			Player.XPos-=TempValue0
		endif
	endif
endsub
