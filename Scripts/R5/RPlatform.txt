	case 1
		Object.Value0-=2
		if Object.Value0<0
			Object.Value0+=512
		endif
		break
	case 2
		Object.Value5+=2
		Object.Value5&=511
		Sin(Object.Value0,Object.Value5)
		Object.Value0>>=2
		Object.Value0+=128
		break
	case 3
		Object.Value5+=2
		Object.Value5&=511
		Sin(Object.Value0,Object.Value5)
		FlipSign(Object.Value0)
		Object.Value0>>=2
		Object.Value0+=384
		Object.Value0&=511
		break
	endswitch
	TempValue1=96
	Cos(Object.Value3,Object.Value0)
	Object.Value3*=TempValue1
	Object.Value3<<=7
	Object.Value3+=Object.XPos
	Object.Value3&=-65536
	Object.Value3-=Object.Value1
	Sin(Object.Value4,Object.Value0)
	Object.Value4*=TempValue1
	Object.Value4<<=7
	Object.Value4+=Object.YPos
	Object.Value4&=-65536
	Object.Value4-=Object.Value2
endsub

	if CheckResult==1
		Player.XPos+=Object.Value3
