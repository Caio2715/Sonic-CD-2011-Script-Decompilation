sub ObjectDraw
	switch Object.State
	case 0
		DrawSprite(0)
		break
	case 1
		DrawSprite(1)
		break
	case 2
		TempValue0 = Object.Frame
		TempValue0 >>= 3
		TempValue0 += 2
		DrawSprite(TempValue0)
		Object.Frame++
		Object.Frame &= 15
		if Object.Value0 < 4
			TempValue0 += 8
			DrawSprite(TempValue0)
		end if
		Object.Value0++
		Object.Value0 &= 7
		break
	case 3
		TempValue0 = Object.Frame
		TempValue0 &= 15
		TempValue0 >>= 3
		TempValue0 += 4
		DrawSprite(TempValue0)
		Object.Frame++

			TempValue0 += 6
			DrawSprite(TempValue0)
		end if
		Object.Value0++
		Object.Value0 &= 7
		if Object.Frame == 48
			Object.Value0 = 0
			Object.Frame = 0
			Object.State = 4
		end if
		break
	case 4
		DrawSprite(0)
		if Object.Value0 < 20
			Object.Value0++
		else
			Object.Value0 = 0
			Object.State = 3
		end if
		break
	case 5
		DrawSprite(1)
		if Object.Value0 < 20
			Object.Value0++
		else
			Object.Value0 = 0
			Object.Frame = 0
			Object.State = 3
		end if
		break
	case 6
		TempValue0 = Object.Frame
		TempValue0 /= 6
		TempValue0 += 6
		DrawSprite(TempValue0)
		Object.Frame++
		Object.Frame %= 24
		TempValue0 = Object.Value0
		TempValue0 /= 6
		TempValue0 += 12
		TempValue1 = Object.Value0
		TempValue1 %= 6
		if TempValue1 < 3
			Object.Direction = FACING_LEFT
			DrawSpriteFX(TempValue0, FX_FLIP, Object.XPos, Object.YPos)
			Object.Direction = FACING_RIGHT
		end if
		Object.Value0++
		Object.Value0 %= 12
		Object.XPos += 65536
		if Object.OutOfBounds == 1
			ResetObjectEntity(Object.EntityNo, TypeName[BlankObject], 0, 0, 0)
		end if
		break
	end switch
end sub
