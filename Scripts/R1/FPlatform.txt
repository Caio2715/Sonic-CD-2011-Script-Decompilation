sub ObjectMain
	if Object.Value0 == 1
		if Object.State == 0
			Object.State = 1
		end if
		if Object.Value1 < 16
			Object.Value1++
			TempValue0 = Object.Value1
			TempValue0 &= 3
			if TempValue0 == 3
				Object.Value2 = 65536
			else
				Object.Value2 = 0
			end if
		else
			Object.Value2 = 0
		end if
	else
		if Object.Value1 > 0
			Object.Value1--
			TempValue0 = Object.Value1
			TempValue0 &= 3
			if TempValue0 == 3
				Object.Value2 = -65536
			else
				Object.Value2 = 0
			end if
		else
			Object.Value2 = 0
		end if
	end if
	Object.Value0 = 0
	if Object.PropertyValue == 0
		switch Object.State
		case 1
			if Object.Value3 < 30
				Object.Value3++
			else
				Object.Value4 = 0
				Object.Value3 = 0
				Object.State = 2
				Object.Priority = 1
			end if
			break
		case 2
			if Object.Value4 < 262144
				Object.Value4 += 8192
			end if
			Object.Value2 += Object.Value4
			if Object.OutOfBounds == 1
				Object.YPos = Object.Value5
				Object.State = 3
				Object.Value4 = 0
			end if
			break
		case 3
			if Object.OutOfBounds == 1
				Object.State = 0
				Object.Priority = 0
			end if
			break
		end switch
	end if
end sub
