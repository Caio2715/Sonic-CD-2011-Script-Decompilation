// Gravity
#alias 0	:	GRAVITY_GROUND
#alias 1	:	GRAVITY_AIR


sub ObjectPlayerInteraction
	PlayerObjectCollision(C_BOX, -16, -8, 16, 24)
	TempValue1 = Object.Value0	// ????
	if CheckResult == true

		if Player.State != PlayerObject_SpinningTop
			Player.Speed = 0
			Object.OnObject = true
			Player.MinRollSpeed = 4096

			if Player.XPos > Object.XPos
				Player.Timer = Player.XPos
				Player.Timer -= Object.XPos

				Object.Rotation = 128
			else
				Player.Timer = Object.XPos
				Player.Timer -= Player.XPos

				Object.Rotation = 384
			end if
			
		end if

		if Player.XPos > Object.XPos
			if Player.Left == true
				Player.Timer -= 4096
			end if

			if Player.Right == true
				Player.Timer += 4096
			end if
		else
			if Player.Left == true
				Player.Timer += 4096
			end if

			if Player.Right == true
				Player.Timer -= 4096
			end if
		end if

		Player.State = PlayerObject_SpinningTop

		Player.Animation = ANI_SPINNING_TOP

		Sin(Player.XPos, Object.Rotation)
		Player.XPos *= Player.Timer
		Player.XPos >>= 9
		Player.XPos += Object.XPos

		TempValue0 = Object.Rotation
		TempValue0 -= 128
		TempValue0 /= 86
		Player.Frame = TempValue0

		Object.Rotation += 16
		if Object.Rotation > 639
			Object.Rotation -= 512
		end if

		if Player.JumpPress == true
			CallFunction(PlayerObject_StartJump)
			Player.Speed = Player.XPos
			Player.Speed -= Object.XPos
			Player.Speed >>= 2

			Player.State = PlayerObject_HandleAir

			Player.Timer = 0
			PlayStageSfx(1, 0)
		end if
	else
		if Object.OnObject == true

			if Player.State == PlayerObject_SpinningTop

				if Player.Gravity == GRAVITY_GROUND
					Player.State = PlayerObject_HandleGround
					Object.OnObject = false
					Player.Timer = 0
				end if

			end if

		end if

	end if
end sub


sub ObjectDraw
	DrawSprite(Object.Frame)
	TempValue0 = Ring.AniCount
	TempValue0 &= 1
	if TempValue0 == 0

		Object.Frame++
		if Object.Frame > 2
			Object.Frame = 0
		end if

	end if
end sub
