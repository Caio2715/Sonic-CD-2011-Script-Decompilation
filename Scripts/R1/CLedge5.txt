			end if
		end if
		break
	case 4
		Object.Value2++
		if Object.Value2 == 12
			Object.Value2 = 0

				Object.Value1 -= 16

sub ObjectPlayerInteraction

		PlayerObjectCollision(C_PLATFORM, Object.Value0, -24, Object.Value1, 16)

			else
				Object.State = 2

		end if

	end if
end sub
