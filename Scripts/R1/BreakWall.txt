sub ObjectMain
	if Object.PropertyValue > 7
		Object.XPos += Object.Value0
		Object.YPos += Object.Value1
		Object.Value1 += 16384
		if Object.OutOfBounds == 1
			Object.Type = TypeName[BlankObject]
		end if
	end if
end sub
