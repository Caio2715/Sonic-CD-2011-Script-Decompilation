sub ObjectMain
	switch Object.State
	case 0
		TempValue0 = 256
		TempValue0 -= Screen.CenterX
		Screen.XOffset = TempValue0
		Object.HelpPage = 0
		CallFunction(HelpMenu_PageLoad)
		PlayMusic(1)
		Object.Timer = 384
		Object.State++
		SetScreenFade(0, 0, 0, 255)
		break

	case 1
		if Object.Timer > 0
			Object.Timer -= 8
		else
			Object.State++
		end if
		SetScreenFade(0, 0, 0, Object.Timer)
		break

	case 2
		if KeyPress[0].Down == true
			if Object.HelpPage < 8
				PlaySfx(27, 0)	//Select SFX
				Object.Timer = 256
				Object.State = 3
			end if
		end if

		if KeyPress[0].Up == true
			if Object.HelpPage > 0
				PlaySfx(27, 0)	//Select SFX
				Object.State = 4
			end if
		end if
		
		if KeyPress[0].ButtonB == true
			PlaySfx(26, 0)
			Object.State = 5
		end if
		break

	case 3
		if Object.Timer > 0
			Cos(TempValue0, Object.Timer)
			TempValue0 >>= 1
			TempValue0 += 256
			TempValue1 = Object.YOffset
			TempValue1 += TempValue0
			Screen.YOffset = TempValue1
			TempValue1 += 256
			TempValue1 >>= 9
			if TempValue1 > Object.HelpPage
				Object.HelpPage++
				if Object.HelpPage == 2
					SetActivePalette(1, 0, Screen.YSize)
				else
					SetActivePalette(0, 0, Screen.YSize)
				end if
				CallFunction(HelpMenu_PageLoad)
			end if

			Object.Timer -= 3
			if Object.Timer < 0
				Object.Timer = 0
			end if
		else
			Object.YOffset += 512
			Screen.YOffset = Object.YOffset
			Object.State = 2
		end if
		break

	case 4
		if Object.Timer < 256
			Cos(TempValue0, Object.Timer)
			TempValue0 >>= 1
			TempValue0 -= 256
			TempValue1 = Object.YOffset
			TempValue1 += TempValue0
			Screen.YOffset = TempValue1
			TempValue1 += 256
			TempValue1 >>= 9
			if TempValue1 < Object.HelpPage
				Object.HelpPage--
				if Object.HelpPage == 2
					SetActivePalette(1, 0, Screen.YSize)
				else
					SetActivePalette(0, 0, Screen.YSize)
				end if
				CallFunction(HelpMenu_PageLoad)
			end if
			
			Object.Timer += 3
			if Object.Timer > 256
				Object.Timer = 256
			end if
		else
			Object.YOffset -= 512
			Screen.YOffset = Object.YOffset
			Object.Timer = 0
			Object.State = 2
		end if
		break

	case 5
		if Object.Timer < 348
			Object.Timer += 8
		else
			Stage.ListPos = 1		// Menu
			Engine.State = ENGINE_EXIT_HIRESMODE
			LoadStage()
		end if
		SetScreenFade(0, 0, 0, Object.Timer)
		break

	end switch
end sub


sub ObjectDraw
	Object.PageTimer--
	if Object.PageTimer < -31
		Object.PageTimer += 32
	end if
	TempValue1 = Object.PageTimer
	TempValue1 >>= 1
	while TempValue1 < Screen.YSize
		TempValue0 = Object.PageTimer
		TempValue0 >>= 1
		while TempValue0 < Screen.XSize
			DrawSpriteScreenXY(0, TempValue0, TempValue1)
			TempValue0 += 128
		loop
		TempValue1 += 128
	loop
end sub


sub ObjectStartup
	LoadSpriteSheet("Menu/HelpBG.gif")
	SpriteFrame(0, 0, 128, 128, 1, 1)
	if Engine.Language == JAPANESE
		Object[59].XPos -= 0x100000
		Object[60].XPos -= 0x100000
		Object[61].XPos -= 0x100000
		Object[62].XPos -= 0x100000
		Object[63].XPos -= 0x100000
		Object[64].XPos += 5242880
		Object[59].YPos += 1572864
		Object[60].YPos += 1572864
		Object[61].YPos += 1572864
		Object[62].YPos += 1572864
		Object[63].YPos += 1572864
		Object[64].YPos -= 3080192
	end if
	LoadPalette("HelpPal1.act", 0, 0, 0, 256)
	LoadPalette("HelpPal2.act", 1, 0, 0, 256)
	LoadTextFont("Data/Game/HelpText.bin")
	if Options.Soundtrack == JP_OST
		SetMusicTrack("JP/DAGarden.ogg", 1, 1)
	else
		SetMusicTrack("US/DAGarden.ogg", 1, 117382)
	end if
	Engine.State = ENGINE_ENTER_HIRESMODE
	Screen.CameraEnabled = 0
	Object[0].Type = TypeName[HelpMenu]
	Object[0].Priority = 1
	Object[0].DrawOrder = 0
end sub
