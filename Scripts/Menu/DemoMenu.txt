sub ObjectMain
#platform: Mobile
	switch Object.State
	case 0
		if Options.PhysicalControls == true
			Object.CurrentButton = -1
		else
			Object.CurrentButton = 0
		end if
		if Object.Fade == 0
			Object.BossUnlockRecord = 9000
			Object.CurrentRecord = SaveRAM[48]	// PPZ1DemoRecord
			Object.CurrentRecord += SaveRAM[49]	// QQZ2DemoRecord
			if Object.CurrentRecord == 0
				SaveRAM[48] = 60000
				SaveRAM[49] = 60000
				Object.CurrentRecord = 120000
			end if
		end if
		if Object.Fade < 256
			Object.Alpha = Object.Fade
			Object.Fade += 6
		else
			Object.Fade = 0
			Object.Alpha = 255
			Object.State++
		end if
		break

	case 1
		if Options.PhysicalControls == true

			if KeyPress.Left == true
				Object.CurrentButton--
				if Object.CurrentButton < 0
					Object.CurrentButton = 2
				end if
				PlaySfx(23, 0) 	//Menu Button SFX
			end if

			if KeyPress.Right == true
				Object.CurrentButton++
				if Object.CurrentButton > 2
					Object.CurrentButton = 0
				end if
				PlaySfx(23, 0) 	//Menu Button SFX
			end if

			CheckTouchRect(0, 0, Screen.XSize, Screen.YSize)
			if CheckResult > -1
				Options.PhysicalControls = false
				Object.CurrentButton = -1
			end if
		else

			CheckTouchRect(0, 0, Screen.XSize, Screen.YSize)
			TempValue3 = CheckResult
			TempValue0 = Screen.CenterX
			TempValue0 -= 113
			TempValue1 = Screen.CenterX
			TempValue1 -= 45
			CheckTouchRect(TempValue0, 67, TempValue1, 157)
			if CheckResult > -1
				Object.CurrentButton = 0
			else
				if TempValue3 < 0
					if Object.CurrentButton == 0
						KeyPress.Start = true
					end if
				else
					if Object.CurrentButton == 0
						Object.CurrentButton = -1
					end if
				end if
			end if

			TempValue0 = Screen.CenterX
			TempValue0 -= 35
			TempValue1 = Screen.CenterX
			TempValue1 += 35
			CheckTouchRect(TempValue0, 67, TempValue1, 157)
			if CheckResult > -1
				Object.CurrentButton = 1
			else
				if TempValue3 < 0
					if Object.CurrentButton == 1
						KeyPress.Start = true
					end if
				else
					if Object.CurrentButton == 1
						Object.CurrentButton = -1
					end if
				end if
			end if

			if Object.CurrentRecord <= Object.BossUnlockRecord
				TempValue0 = Screen.CenterX
				TempValue0 += 45
				TempValue1 = Screen.CenterX
				TempValue1 += 113
				CheckTouchRect(TempValue0, 67, TempValue1, 157)
				if CheckResult > -1
					Object.CurrentButton = 2
				else
					if TempValue3 < 0
						if Object.CurrentButton == 2
							KeyPress.Start = true
						end if
					else
						if Object.CurrentButton == 2
							Object.CurrentButton = -1
						end if
					end if
				end if
			end if
			CallFunction(MenuHeading_CheckTouchBack)
			if KeyPress.Left == true
				Object.CurrentButton = 1
				Options.PhysicalControls = true
			end if
			if KeyPress.Right == true
				Object.CurrentButton = 0
				Options.PhysicalControls = true
			end if

		end if
		CheckResult = false
		if Object.CurrentButton > -1
			if KeyPress.Start == true
				if Object.CurrentButton == 2
					if Object.CurrentRecord <= Object.BossUnlockRecord
						CheckResult = true
					end if
				else
					CheckResult = true
				end if
			end if
			if KeyPress.ButtonA == true
				if Object.CurrentButton == 2
					if Object.CurrentRecord <= Object.BossUnlockRecord
						CheckResult = true
					end if
				else
					CheckResult = true
				end if
			end if
		end if
		if CheckResult == true
			Object.State++
			StopMusic()
			PlaySfx(27, 0)	//Select SFX
		end if
		if KeyPress.ButtonB == true
			Object.Fade = 248
			Object.State = DEMOMENU_BACK
			PlayStageSfx(1, 0)
		end if
		break

	case DEMOMENU_LEVEL_SELECTED
		if Object.AnimationTimer < 30
			Object.AnimationTimer++
		else
			Object.AnimationTimer = 0
			Object.State = DEMOMENU_LEVEL_LOAD
		end if
		break

	case DEMOMENU_BACK
		if Object.Fade > 0
			Object.Alpha = Object.Fade
			Object.Fade -= 8
		else
			Object.Type = TypeName[Blank Object]
			Object.MWState = 3
			//All of this is MenuButton Objects
			Object[34].State = 1
			Object[35].State = 1
			Object[36].State = 1
			Object[37].State = 1
			Object[38].State = 1
			Object[39].State = 1
			Object[40].State = 1
			Object[41].State = 1
			Object[42].State = 1
			//End
			Object.MHState = 1
			Object.MHPropertyValue = 5
		end if
		break

	case DEMOMENU_LEVEL_LOAD
		if Object.Fade < 320
			Object.Fade += 8
		else
			Options.GameMode = GAMEMODE_DEMOMODE
			Player.Score = 0
			Player.Lives = 3
			Player.ScoreBonus = 50000
			Transporter_Destroyed = 0
			MetalSonic_Destroyed = 0
			TimeAttack.Round = Object.CurrentButton
			Stage.ActiveList = REGULAR_STAGE
			switch Object.CurrentButton
			case 0
				Stage.ListPos = 0		//Palmtree Panic Zone 1 Present
				Good_Future = 0
				break
			case 1
				Stage.ListPos = 34		//Quartz Quadrant Zone 2 Present
				Good_Future = 0
				break
			case 2
				Stage.ListPos = 38		//Quartz Quadrant Zone 3 Good Future
				Good_Future = 1
				break
			end switch
			LoadStage()
		end if
		SetScreenFade(0, 0, 0, Object.Fade)
		break

	end switch
#endplatform
end sub


sub ObjectDraw
#platform: Mobile
	TempValue0 = Screen.CenterX
	TempValue0 -= 113
	DrawSpriteScreenFX(0, FX_INK, TempValue0, 57)
	TempValue2 = 33
	TempValue3 = 267
	CallFunction(DemoMenu_StagePortraitMovement)
	DrawSpriteScreenFX(10, FX_INK, TempValue0, 57)
	DrawSpriteScreenFX(6, FX_INK, TempValue0, 134)
	TempValue1 = SaveRAM[48]	// PPZ1DemoRecord
	CallFunction(DemoMenu_StageTimeRecord)
	TempValue0 += 78
	DrawSpriteScreenFX(1, FX_INK, TempValue0, 57)
	TempValue2 = 33
	TempValue3 = 396
	CallFunction(DemoMenu_StagePortraitMovement)
	DrawSpriteScreenFX(10, FX_INK, TempValue0, 57)
	DrawSpriteScreenFX(6, FX_INK, TempValue0, 134)
	TempValue1 = SaveRAM[49]	// QQZ2DemoRecord
	CallFunction(DemoMenu_StageTimeRecord)
	TempValue0 += 78
	DrawSpriteScreenFX(2, FX_INK, TempValue0, 57)
	TempValue2 = 321
	TempValue3 = 399
	CallFunction(DemoMenu_StagePortraitMovement)
	DrawSpriteScreenFX(10, FX_INK, TempValue0, 57)
	Object.Rotation += 129
	Object.Rotation &= 511
	TempValue0 = Screen.CenterX
	TempValue1 = Object.AnimationTimer
	TempValue1 &= 3
	TempValue1 >>= 1
	TempValue1 += 7
	switch Object.CurrentButton
	case 0
		TempValue0 -= 113
		DrawSpriteScreenFX(TempValue1, FX_INK, TempValue0, 57)
		break

	case 1
		TempValue0 -= 35
		DrawSpriteScreenFX(TempValue1, FX_INK, TempValue0, 57)
		break

	case 2
		TempValue0 += 43
		DrawSpriteScreenFX(TempValue1, FX_INK, TempValue0, 57)
		break

	end switch
	if Object.CurrentRecord > Object.BossUnlockRecord
		TempValue0 = Screen.CenterX
		TempValue0 += 97
		DrawSpriteScreenFX(9, FX_INK, TempValue0, 128)
		TempValue0 = Screen.CenterX
		TempValue0 -= 113
		DrawSpriteScreenFX(3, FX_INK, TempValue0, 155)
		DrawSpriteScreenFX(4, FX_INK, TempValue0, 155)
		TempValue0 += 44
		Object.Frame = 21
		TempValue1 = Object.CurrentRecord
		CallFunction(DemoMenu_StageTotalTimeRecord)
		TempValue0 += 114
		DrawSpriteScreenFX(5, FX_INK, TempValue0, 154)
		TempValue0 += 11
		Object.Frame = 31
		TempValue1 = Object.BossUnlockRecord
		CallFunction(DemoMenu_StageTotalTimeRecord)
	else
		TempValue0 = Screen.CenterX
		TempValue0 -= 113
		DrawSpriteScreenFX(3, FX_INK, TempValue0, 155)
		DrawSpriteScreenFX(4, FX_INK, TempValue0, 155)
		TempValue0 += 44
		Object.Frame = 21
		TempValue1 = Object.CurrentRecord
		CallFunction(DemoMenu_StageTotalTimeRecord)
	end if
#endplatform
end sub
