					Object.Value1 = 1
					Options.PhysicalControls = 1
				end if
				if KeyPress.Down == 1
					Object.Value1 = 0
					Options.PhysicalControls = 1
				end if

			CheckResult = 0
			if Object.Value1 > -1
				if KeyPress.Start == 1
					CheckResult = 1
				end if

					CheckResult = 1
				end if
			end if
			if CheckResult == 1
				PlaySfx(27, 0)
				Object.State++
			end if
			break
		case 1
			if Object.AnimationTimer < 30
				Object.AnimationTimer++
			else
				Object.AnimationTimer = 0
				Object.State = 0

				case 0
					EngineCallback(9)
					break
				case 1

				end switch

			end if
			break
		end switch
	else
		if Options.PhysicalControls == 0
			Object.Value1 = -1
		end if
