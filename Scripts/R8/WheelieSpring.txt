sub ObjectMain
	if Object.State == 0
		if Object.Value0 < 96
			Object.Value0++
			Object.XPos += 65536
			ObjectTileGrip(0, 8, 15, 0)
			if CheckResult == 0
				Object.State = 1
				Object.Value0 = -64
			else
				ObjectTileCollision(1, 40, -16, 0)
				if CheckResult == 1
					Object.State = 1
					Object.Value0 = -64
				end if
			end if
			ObjectTileGrip(0, 0, 15, 0)
		else
			Object.State = 1
			Object.Value0 = -64
		end if
	else
		if Object.Value0 < 96
			Object.Value0++
			Object.XPos -= 65536
			ObjectTileGrip(0, -8, 15, 0)
			if CheckResult == 0
				Object.State = 0
				Object.Value0 = -64
			else
				ObjectTileCollision(3, -40, -16, 0)
				if CheckResult == 1
					Object.State = 0
					Object.Value0 = -64
				end if
			end if
			ObjectTileGrip(0, 0, 15, 0)
		else
			Object.State = 0
			Object.Value0 = -64
		end if
	end if
end sub


sub ObjectPlayerInteraction
	if Player.State != PlayerObject_Blank
		if Player.YVelocity >= 0
			PlayerObjectCollision(C_BOX, -16, -16, 16, 16)
			PlayerObjectCollision(C_TOUCH, -15, -18, 15, -12)
			if CheckResult == 1
				Object.Frame = 1
				Player.State = PlayerObject_HandleAir
				Player.Gravity = 1
				if Object.PropertyValue == 0
					Player.YVelocity = -1048576
				else
					Player.YVelocity = -655360
				end if
				Player.AnimationCheck = ANI_WALKING
				if Player.Animation == ANI_RUNNING
					Player.AnimationCheck = ANI_RUNNING
				end if
				if Player.Animation == ANI_PEELOUT
					Player.AnimationCheck = ANI_PEELOUT
				end if
				Player.Animation = 11
				Player.Timer = 0
				PlaySfx(11, 0)
			end if
		end if
		if Player.Gravity == 0
			PlayerObjectCollision(C_BOX, -16, -20, 16, 16)
			PlayerObjectCollision(C_TOUCH, -15, -18, 15, -12)
			if CheckResult == 1
				Object.Frame = 1
				Player.State = PlayerObject_HandleAir
				Player.Gravity = 1
				if Object.PropertyValue == 0
					Player.YVelocity = -1048576
				else
					Player.YVelocity = -655360
				end if
				Player.Animation = 11
				Player.Timer = 0
				PlaySfx(11, 0)
			end if
		end if
	end if
end sub


sub ObjectDraw
	if Object.PropertyValue == 0
		if Object.Frame == 0
			DrawSprite(0)
		else
			Object.Frame++
			TempValue0 = Object.Frame
			TempValue0 >>= 1
			switch TempValue0
			case 0
			case 1
				DrawSprite(0)
				break
			case 2
			case 6
				DrawSprite(2)
				break
			case 3
			case 4
			case 5
				DrawSprite(4)
				break
			end switch
			if Object.Frame > 12
				Object.Frame = 0
			end if
		end if
	else
		if Object.Frame == 0
			DrawSprite(1)
		else
			Object.Frame++
			TempValue0 = Object.Frame
			TempValue0 >>= 1
			switch TempValue0
			case 0
			case 1
				DrawSprite(1)
				break
			case 2
			case 6
				DrawSprite(3)
				break
			case 3
			case 4
			case 5
				DrawSprite(5)
				break
			end switch
			if Object.Frame > 12
				Object.Frame = 0
			end if
		end if
	end if
	TempValue0 = Object.Value0
	TempValue0 += 64
	TempValue0 %= 18
	TempValue0 /= 9
	TempValue0 += 6
	DrawSprite(TempValue0)
end sub
