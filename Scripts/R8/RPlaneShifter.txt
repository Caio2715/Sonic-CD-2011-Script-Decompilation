sub ObjectMain
	if Player.ObjectInteraction == 1
		if Object.PropertyValue == 0
			switch Object.State
			case 1
				if Object.Value1 < 256
					Object.Value1 += 8
				else
					Object.State = 0
					Player.State = PlayerObject_HandleGround
					Player.Frame = 0
					Player.DrawOrder = 2
					Player.CollisionPlane = 1
				end if
				TempValue0 = Object.Value1
				TempValue0 >>= 6
				switch TempValue0
				case 0
					Player.Frame = 1
					break
				case 1
					Player.Frame = 2
					break
				case 2
					Player.Frame = 3
					break
				case 3
					Player.Frame = 0
					Player.Direction = FACING_LEFT
					break
				end switch
				Object.AnimationTimer++
				Object.AnimationTimer %= 6
				break
			case 2
				if Object.Value1 > 0
					Object.Value1 -= 8
				else
					Object.State = 0
					Player.State = PlayerObject_HandleGround
					Player.Frame = 0
					Player.DrawOrder = 4
					Player.CollisionPlane = 0
				end if
				TempValue0 = Object.Value1
				TempValue0 >>= 6
				switch TempValue0
				case 0
					Player.Frame = 0
					Player.Direction = FACING_LEFT
					break
				case 1
					Player.Frame = 4
					break
				case 2
					Player.Frame = 0
					break
				case 3
					Player.Frame = 1
					break
				end switch
				Object.AnimationTimer++
				Object.AnimationTimer %= 6
				break
			case 3
				if Object.Value1 > 256
					Object.Value1 -= 8
				else
					Object.State = 0
					Player.State = PlayerObject_HandleGround
					Player.Frame = 0
					Player.DrawOrder = 2
					Player.CollisionPlane = 1
				end if
				TempValue0 = 512
				TempValue0 -= Object.Value1
				TempValue0 >>= 6
				switch TempValue0
				case 0
					Player.Frame = 1
					break
				case 1
					Player.Frame = 2
					break
				case 2
					Player.Frame = 3
					break
				case 3
					Player.Frame = 0
					Player.Direction = FACING_RIGHT
					break
				end switch
				Object.AnimationTimer++
				Object.AnimationTimer %= 6
				break
			case 4
				if Object.Value1 < 512
					Object.Value1 += 8
				else
					Object.Value1 = 0
					Object.State = 0
					Player.State = PlayerObject_HandleGround
					Player.Frame = 0
					Player.DrawOrder = 4
					Player.CollisionPlane = 0
				end if
				TempValue0 = 512
				TempValue0 -= Object.Value1
				TempValue0 >>= 6
				switch TempValue0
				case 0
					Player.Frame = 0
					Player.Direction = FACING_RIGHT
					break
				case 1
					Player.Frame = 4
					break
				case 2
					Player.Frame = 0
					break
				case 3
					Player.Frame = 1
					break
				end switch
				Object.AnimationTimer++
				Object.AnimationTimer %= 6
				break
			end switch
			Sin(Object[+1].XPos, Object.Value1)
			Object[+1].XPos *= 3072
			Object[+1].XPos += Object.XPos
			Object[+1].YPos = Object.YPos
			if Object.Value1 > 128
				if Object.Value1 < 384
					Object[+1].DrawOrder = 1
				else
					Object[+1].DrawOrder = 4
				end if
			else
				Object[+1].DrawOrder = 4
			end if
			TempValue0 = Object.Value1
			TempValue0 += 170

sub ObjectPlayerInteraction
	if Object.PropertyValue == 0
		PlayerObjectCollision(C_PLATFORM, -40, -8, 40, 24)
		TempValue1 = CheckResult
		PlayerObjectCollision(C_BOX, -40, -184, 40, -168)
		PlayerObjectCollision(C_TOUCH, -4, -48, 4, 8)
		if CheckResult == 1
			if Object.Value2 == 0
				if TempValue1 == 1
					Object.Value2 = 1
					Player.State = PlayerObject_Blank
					Player.Animation = ANI_STOPPED
					Player.XPos = Object.XPos
					Player.Speed = 0
					Player.XVelocity = 0
					Player.YVelocity = 0
					Player.TrackScroll = 0
					if Player.Direction == FACING_RIGHT
						if Player.CollisionPlane == 0
							Object.State = 1
							Object.Value1 = 0
						else
							Object.State = 2
							Object.Value1 = 256
						end if
					else
						if Player.CollisionPlane == 0
							Object.State = 3
							Object.Value1 = 512
						else
							Object.State = 4
							Object.Value1 = 256
						end if
					end if
				end if
			end if
		else
			Object.Value2 = 0
		end if
	end if
end sub


sub ObjectDraw
	if Object.PropertyValue == 0
		TempValue0 = Object.AnimationTimer
		TempValue0 /= 3
		DrawSprite(TempValue0)
		TempValue0 += 2
		DrawSprite(TempValue0)
	else
		DrawSprite(4)
		DrawSprite(5)
		DrawSprite(6)
		DrawSprite(7)
		DrawSprite(8)
	end if
end sub
