sub ObjectMain
	if Object.Value5 == 0
		Object.Value3 += Object.Value6
		Object.Value3 &= 511
		if Object.PropertyValue == 1
			Object.Value3 += 256
		end if
		Sin(Object.Value2, Object.Value3)
		Object.Value2 *= Object.Value4
		Object.Value2 += Object.YPos
		Object.Value2 &= -0x10000
		Object.Value2 -= Object.Value1
	else
		switch Object.State
		case 0
			Object.Value3 += 4
			Object.Value3 &= 511
			if Object.Value3 == 128
				Object.State++
			end if
			break
		case 1
			if Object.Value0 < 64
				Object.Value0++
			else
				Object.Value0 = 0
				Object.State++
			end if
			break
		case 2
			Object.Value3 += 4
			Object.Value3 &= 511
			if Object.Value3 == 384
				Object.State++
			end if
			break
		case 3
			if Object.Value0 < 64
				Object.Value0++
			else
				Object.Value0 = 0
				Object.State = 0
			end if
			break
		end switch
		Sin(Object.Value2, Object.Value3)
		Object.Value2 *= Object.Value4
		Object.Value2 += Object.YPos
		Object.Value2 &= -0x10000
		Object.Value2 -= Object.Value1
	end if
end sub


sub ObjectPlayerInteraction
	TempValue0 = Object.YPos
	TempValue1 = Player.YPos
	Object.YPos = Object.Value1
	if Object.Value5 == 0
		PlayerObjectCollision(C_BOX, -32, -32, 32, 32)
		if CheckResult == 1
			Player.YPos += Object.Value2
		else
			if CheckResult == 4
				if Player.Gravity == 0
					if Object.Value2 > 0
						CallFunction(Player_Kill)
					else
						Player.YPos = TempValue1
					end if
				end if
			end if
		end if
	else
		PlayerObjectCollision(C_BOX, -32, -96, 32, 96)
		if CheckResult == 1
			Player.YPos += Object.Value2
			if Object.Value3 == 320
				
#platform: Use_Origins
				Player.State = Player_State_Air_NoDropDash
#endplatform

#platform: Use_Standalone
				Player.State = Player_State_Air
#endplatform
				Player.Timer = 0
				Player.YVelocity = -0x100000
				Player.Animation = ANI_BOUNCING
				Player.Gravity = 1
				Player.AnimationReserve = ANI_WALKING
			end if
		else
			if CheckResult == 4
				if Player.Gravity == 0
					if Object.Value2 > 0
						CallFunction(Player_Kill)
					else
						Player.YPos = TempValue1
					end if
				end if
			end if
		end if
	end if
	Object.YPos = TempValue0
end sub


sub ObjectDraw
	Sin(Object.Value1, Object.Value3)
	Object.Value1 *= Object.Value4
	Object.Value1 += Object.YPos
	Object.Value1 &= -0x10000
	DrawSpriteXY(0, Object.XPos, Object.Value1)
	if Object.Value5 == 1
		DrawSpriteXY(1, Object.XPos, Object.Value1)
		DrawSpriteXY(2, Object.XPos, Object.Value1)
	end if
end sub

	ArrayPos0 = 32
	while ArrayPos0 < 1056
		if Object[ArrayPos0].Type == TypeName[CrushPlatform]
			Object[ArrayPos0].Value6 = 2
			if Object[ArrayPos0].PropertyValue == 0
				Object[ArrayPos0].Value4 = 0x1000
			else
				Object[ArrayPos0].Value4 = 0x1800
			end if

			end if

				Object[ArrayPos0].Value6 = 6
			end if
		end if

	loop
end sub
