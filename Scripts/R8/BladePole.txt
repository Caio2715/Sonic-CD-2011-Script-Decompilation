sub ObjectMain
	switch Object.State
	case 0
		ObjectTileGrip(0, 0, 48, 0)
		if CheckResult == 1
			Object.State = 1
			Object.Value0 = -64
		else
			Object.YPos += 65536
		end if
		break
	case 1
		if Object.Value0 < 96
			Object.Value0++
			Object.XPos += 65536
			ObjectTileGrip(0, 0, 48, 0)
			if CheckResult == 0
				Object.State = 2
				Object.Value0 = -64
			end if
		else
			Object.State = 2
			Object.Value0 = -64
		end if
		break
	case 2
		if Object.Value0 < 96
			Object.Value0++
			Object.XPos -= 65536
			ObjectTileGrip(0, 0, 48, 0)
			if CheckResult == 0
				Object.State = 1
				Object.Value0 = -64
			end if
		else
			Object.State = 1
			Object.Value0 = -64
		end if
		break
	end switch
	switch Object.Frame
	case 0
	case 2
		if Object.AnimationTimer < 4
			Object.AnimationTimer++
		else
			Object.AnimationTimer = 0
			Object.Frame++
		end if
		break
	case 1
		if Object.AnimationTimer < 72
			Object.AnimationTimer++
		else
			Object.AnimationTimer = 0
			Object.Frame++
		end if
		break
	case 3
		if Object.AnimationTimer < 72
			Object.AnimationTimer++
		else
			Object.AnimationTimer = 0
			Object.Frame = 0
		end if
		break
	end switch
end sub


sub ObjectPlayerInteraction
	switch Object.Frame
	case 1
		PlayerObjectCollision(C_TOUCH, -28, -48, 28, -40)
		if CheckResult == 1
			CallFunction(PlayerObject_Hit)
		end if
		break
	case 3
		PlayerObjectCollision(C_TOUCH, -28, 16, 28, 24)
		if CheckResult == 1
			CallFunction(PlayerObject_Hit)
		end if
		break
	end switch
end sub


sub ObjectDraw
	TempValue0 = Object.Value0
	TempValue0 += 64
	TempValue0 %= 18
	TempValue0 /= 9
	TempValue0 += 16
	DrawSprite(TempValue0)
	switch Object.Frame
	case 0
	case 2
		DrawSprite(0)
		Object.Direction = 2
		DrawSpriteFX(2, FX_FLIP, Object.XPos, Object.YPos)
		break
	case 1
		TempValue0 = Object.AnimationTimer
		TempValue0 /= 3
		TempValue0 %= 6
		TempValue0 += 4
		DrawSprite(TempValue0)
		DrawSpriteFX(3, FX_FLIP, Object.XPos, Object.YPos)
		break
	case 3
		TempValue0 = Object.AnimationTimer
		TempValue0 /= 3
		TempValue0 %= 6
		TempValue0 += 10
		DrawSprite(1)
		DrawSpriteFX(TempValue0, FX_FLIP, Object.XPos, Object.YPos)
		break
	end switch
end sub
