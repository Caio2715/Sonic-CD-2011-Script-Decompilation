// Aliases
#alias Object.Value0	:	Object.Timer

// Property Values
#alias 0	:	FADE_TO_BOSS
#alias 1	:	FADE_TO_LEVEL
#alias 2	:	SSZEGGMAN
#alias 3	:	SSZAMY

// Priority
#alias 1	:	PRIORITY_ACTIVE


sub ObjectMain
	
#platform: Use_Origins
	// Take some values and print them?
	// they left debug stuff in lol
	game.callbackParam0 = 100					// Max timer value?
	game.callbackParam1 = Object.PropertyValue	// Get Fade Type
	game.callbackParam2 = Object.Timer			// Get current music timer
	EngineCallback(NOTIFY_DEBUGPRINT)
#endplatform

	switch Object.PropertyValue
	case FADE_TO_BOSS
		if Object.Timer < 100
			Object.Timer++
			Music.Volume--
		else
			PlayMusic(4)
			Object.Type = TypeName[Blank Object]
		end if
		break

	case FADE_TO_LEVEL
		if Object.Timer < 100
			Object.Timer++
			Music.Volume--
		else
			PlayMusic(0)
			Object.Type = TypeName[Blank Object]
		end if
		break

	case SSZEGGMAN
		if Object.Timer < 50
			Object.Timer++
			Music.Volume -= 2
		else
			PlayMusic(0)
			Object.Type = TypeName[Blank Object]
		end if
		break

	case SSZAMY
		if Object.Timer < 100
			Object.Timer++
			Music.Volume--
		else
		
#platform: Use_Origins
			if game.playMode != BOOT_PLAYMODE_BOSSRUSH
				PlayMusic(1)
			endif
#endplatform

#platform: Use_Standalone
			PlayMusic(1)
#endplatform
			
			Object.Type = TypeName[Blank Object]
			
			Object[30].Type		 = TypeName[ActFinish]
			Object[30].DrawOrder = 6
			Object[30].Priority  = PRIORITY_ACTIVE
		end if
		break

	end switch
end sub
