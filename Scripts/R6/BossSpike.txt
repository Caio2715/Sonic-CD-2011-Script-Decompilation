sub ObjectMain
	if Object.State == 0
		Object.YPos += 229376
		ObjectTileCollision(0, 0, 12, 0)
		if CheckResult == 1
			Object.State = 1
		end if
	else
		Object.Value0++
		if Object.Value0 == 64
			Object.Type = TypeName[BlankObject]
		end if
	end if
end sub


sub ObjectPlayerInteraction
	if Object.State == 0
		PlayerObjectCollision(C_TOUCH, -4, -12, 4, 12)
		if CheckResult == 1
			CallFunction(PlayerObject_Hit)
		end if
	end if
end sub


sub ObjectDraw
	TempValue0 = Object.Value0
	TempValue0 &= 7
	if TempValue0 < 4
		DrawSprite(0)
	end if
end sub

end sub
