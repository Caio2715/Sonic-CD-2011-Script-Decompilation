sub ObjectMain
	if Object.State == 0
		if Object.AnimationTimer < 64
			Object.AnimationTimer++
			CallFunction(Object.Rotation)
		else
			Object.AnimationTimer = 0
			Object.State = 1
			Object.Direction++
			if Object.Direction > 14
				Object.Direction = 3
			end if
		end if
	else
		if Object.AnimationTimer < 30
			Object.AnimationTimer++
		else
			Object.AnimationTimer = 0
			Object.State = 0
		end if
	end if
end sub

sub ObjectStartup
	LoadSpriteSheet("R6/Objects.gif")
	SpriteFrame(-16, -16, 32, 32, 173, 1)
	ArrayPos0 = 32
	while ArrayPos0 < 1056
		if Object[ArrayPos0].Type == TypeName[MovingBlocks]
			Object[ArrayPos0].BlockXPos_1 = Object[ArrayPos0].XPos
			Object[ArrayPos0].BlockYPos_1 = Object[ArrayPos0].YPos
			Object[ArrayPos0].BlockXPos_2 = Object[ArrayPos0].XPos
			Object[ArrayPos0].BlockYPos_2 = Object[ArrayPos0].YPos
			Object[ArrayPos0].BlockXPos_3 = Object[ArrayPos0].XPos
			Object[ArrayPos0].BlockYPos_3 = Object[ArrayPos0].YPos
			Object[ArrayPos0].BlockXPos_4 = Object[ArrayPos0].XPos
			Object[ArrayPos0].BlockYPos_4 = Object[ArrayPos0].YPos
			switch Object[ArrayPos0].PropertyValue
			case 0
				Object[ArrayPos0].Rotation = MovingBlocks_MoveType_1
				break
			case 1
				Object[ArrayPos0].Rotation = MovingBlocks_MoveType_4
				break
			case 3
				Object[ArrayPos0].Rotation = MovingBlocks_MoveType_2
				break
			case 4
				Object[ArrayPos0].Rotation = MovingBlocks_MoveType_3
				break
			end switch
		end if
		ArrayPos0++
	loop
end sub
