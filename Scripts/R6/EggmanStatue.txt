sub ObjectMain
	switch Object.State
	case EGGMANSTATUE_DESTROYED
		TempValue0 = Object.Timer
		TempValue0 &= 3
		if TempValue0 == 0
			Rand(TempValue0, 32)
			TempValue0 -= 16
			TempValue0 <<= 16
			TempValue0 += Object.XPos

			Rand(TempValue1, 48)
			TempValue1 -= 24
			TempValue1 <<= 16
			TempValue1 += Object.YPos
			
			CreateTempObject(TypeName[Explosion], 0, TempValue0, TempValue1)
			PlaySfx(22, 0)
		end if
		
		Object.Timer++
		if Object.Timer == 40
			Object.Timer = 0
			Object.State++
		end if
		break

	case EGGMANSTATUE_BOMB_DROP
		Object.Timer++
		if Object.Timer == 60
			Object.Timer = 0
			CallFunction(EggmanStatue_HandleBombDrop)
			Object.Type = TypeName[Blank Object]
		end if
		break

	end switch
end sub


sub ObjectPlayerInteraction
	if Object.State == EGGMANSTATUE_INACTIVE
		if Player.Animation == ANI_JUMPING
			PlayerObjectCollision(C_TOUCH, -16, -24, 16, 24)
			if CheckResult == true
				Object.State = EGGMANSTATUE_DESTROYED
			end if
		end if
	end if
end sub


sub ObjectDraw
	if Object.State < EGGMANSTATUE_BOMB_DROP
		DrawSprite(0)
	end if
end sub
