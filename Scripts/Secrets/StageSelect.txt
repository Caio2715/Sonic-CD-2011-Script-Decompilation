sub ObjectMain
	switch Object.State
	case 0
		if Object.Value0 < 8
			Object.Value0++
		else
			Object.Value0 = 254
			Object.State++
		end if
		SetScreenFade(0, 0, 0, 255)
		break
	case 1
		if Object.Value0 > 0
			SetScreenFade(0, 0, 0, Object.Value0)
			Object.Value0 -= 8
		else
			Object.State++
		end if
		break
	case 2
		if KeyPress.Up == true
			Object.Value2 = 0
			Object.Value1--
			if Object.Value1 < 0
				Object.Value1 = 69
			end if
		else
			if KeyPress.Down == true
				Object.Value2 = 0
				Object.Value1++
				if Object.Value1 > 69
					Object.Value1 = 0
				end if
			end if
		end if
		if KeyDown.Up == true
			Object.Value2++
			if Object.Value2 == 8
				Object.Value2 = 0
				Object.Value1--
				if Object.Value1 < 0
					Object.Value1 = 69
				end if
			end if
		else
			if KeyDown.Down == true
				Object.Value2++
				if Object.Value2 == 8
					Object.Value2 = 0
					Object.Value1++
					if Object.Value1 > 69
						Object.Value1 = 0
					end if
				end if
			end if
		end if
		if KeyPress.Start == true
			Object.State = 3
		end if
		if KeyPress.ButtonB == true
			Object.State = 4
		end if
#platform: Mobile
		TempValue0 = Screen.CenterX
		TempValue0 -= 138
		TempValue1 = Screen.CenterX
		TempValue1 -= 76
		CheckTouchRect(TempValue0, 20, TempValue1, 72)
		if CheckResult > -1
			KeyDown.Up = 1
			Object.Value2++
			if Object.Value2 == 8
				Object.Value2 = 0
				Object.Value1--
				if Object.Value1 < 0
					Object.Value1 = 69
				end if
			end if
		end if
		CheckTouchRect(TempValue0, 162, TempValue1, 214)
		if CheckResult > -1
			KeyDown.Down = 1
			Object.Value2++
			if Object.Value2 == 8
				Object.Value2 = 0
				Object.Value1++
				if Object.Value1 > 69
					Object.Value1 = 0
				end if
			end if
		end if
		if Engine.Message == 2
			Object.Value3 = 0
		end if
		TempValue0 = Screen.CenterX
		TempValue0 += 28
		TempValue1 = Screen.CenterX
		TempValue1 += 68
		CheckTouchRect(TempValue0, 162, TempValue1, 214)
		if CheckResult > -1
			Object.Value3 = 1
		else
			CheckTouchRect(0, 0, Screen.XSize, Screen.YSize)
			if CheckResult < 0
				if Object.Value3 == 1
					Object.Value3 = 0
					PlaySfx(27, 0)	//Select SFX
					Object.State = 4
				end if
			else
				if Object.Value3 == 1
					Object.Value3 = 0
				end if
			end if
		end if
		TempValue0 += 48
		TempValue1 += 48
		CheckTouchRect(TempValue0, 162, TempValue1, 214)
		if CheckResult > -1
			Object.Value3 = 2
		else
			CheckTouchRect(0, 0, Screen.XSize, Screen.YSize)
			if CheckResult < 0
				if Object.Value3 == 2
					Object.Value3 = 0
					PlaySfx(27, 0)	//Select SFX
					Object.State = 3
				end if
			else
				if Object.Value3 == 2
					Object.Value3 = 0
				end if
			end if
		end if
#endplatform
		break
	case 3
		if Object.Value0 < 288
			Object.Value0 += 8
		else
			Options.GameMode = 0
#platform: Mobile
			Player.Lives = 3
#endplatform
			Stage.ActiveList = REGULAR_STAGE
			Stage.ListPos = Object.Value1
			LoadStage()
		end if
		TempValue0 = Object.Value0
		if TempValue0 > 255
			TempValue0 = 255
		end if
		SetScreenFade(0, 0, 0, TempValue0)
		break
	case 4
		if Object.Value0 < 288
			Object.Value0 += 8
		else
			Options.GameMode = 0
			Stage.ActiveList = PRESENTATION_STAGE
			Stage.ListPos = 1
			LoadStage()
		end if
		TempValue0 = Object.Value0
		if TempValue0 > 255
			TempValue0 = 255
		end if
		SetScreenFade(0, 0, 0, TempValue0)
		break
	end switch
end sub


sub ObjectDraw
	TempValue0 = 0
	while TempValue0 < 240
		TempValue1 = 0
		while TempValue1 < Screen.XSize
			DrawSpriteScreenXY(15, TempValue1, TempValue0)
			TempValue1 += 224
		loop
		TempValue0 += 48
	loop
	DrawSpriteScreenXY(16, Screen.CenterX, 80)
	TempValue0 = Screen.CenterX
	TempValue0 -= 48
	TempValue1 = Object.Value1
	TempValue1 /= 10
	switch TempValue1
	case 0
		DrawSpriteScreenXY(1, TempValue0, 112)
		break
	case 1
		DrawSpriteScreenXY(3, TempValue0, 112)
		break
	case 2
		DrawSpriteScreenXY(4, TempValue0, 112)
		break
	case 3
		DrawSpriteScreenXY(5, TempValue0, 112)
		break
	case 4
		DrawSpriteScreenXY(6, TempValue0, 112)
		break
	case 5
		DrawSpriteScreenXY(7, TempValue0, 112)
		break
	case 6
		DrawSpriteScreenXY(8, TempValue0, 112)
		break
	end switch
	TempValue0 += 8
	DrawSpriteScreenXY(14, TempValue0, 112)
	TempValue0 += 8
	TempValue1 = Object.Value1
	TempValue1 %= 10
	TempValue1 >>= 2
	TempValue1++
	DrawSpriteScreenXY(TempValue1, TempValue0, 112)
	TempValue0 += 8
	DrawSpriteScreenXY(14, TempValue0, 112)
	TempValue0 += 8
	TempValue1 = Object.Value1
	TempValue1 %= 10
	switch TempValue1
	case 0
	case 4
		DrawSpriteScreenXY(10, TempValue0, 112)
		break
	case 1
	case 5
		DrawSpriteScreenXY(11, TempValue0, 112)
		break
	case 2
	case 6
	case 8
		DrawSpriteScreenXY(12, TempValue0, 112)
		break
	case 3
	case 7
	case 9
		DrawSpriteScreenXY(13, TempValue0, 112)
		break
	end switch
#platform: Mobile
	TempValue0 -= 96
	if KeyDown.Up == true
		DrawSpriteScreenXY(18, TempValue0, 48)
	else
		DrawSpriteScreenXY(17, TempValue0, 48)
	end if
	if KeyDown.Down == true
		DrawSpriteScreenXY(20, TempValue0, 192)
	else
		DrawSpriteScreenXY(19, TempValue0, 192)
	end if
	TempValue0 += 160
	if Object.Value3 == 1
		DrawSpriteScreenXY(24, TempValue0, 192)
	else
		DrawSpriteScreenXY(23, TempValue0, 192)
	end if
	TempValue0 += 48
	if Object.Value3 == 2
		DrawSpriteScreenXY(22, TempValue0, 192)
	else
		DrawSpriteScreenXY(21, TempValue0, 192)
	end if
#endplatform
end sub
