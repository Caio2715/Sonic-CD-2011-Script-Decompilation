// Aliases
#alias Object.Value0			:	Object.DeformTimer
#alias Object.Value2			:	Object.BumperOscillation
#alias Object.Value3			:	Object.TurnDirection

// Hud Aliases
#alias Object[24].PropertyValue	:	HUD.TimePeriod

// Time Periods
#alias 1	:	PROPVAL_PAST

#alias 1	:	GRAVITY_AIR

sub ObjectMain
	if Object.TurnDirection == false
		if Object.BumperOscillation < 0x300000		// 0x380000
			Object.BumperOscillation += 0x10000		// 0x10000
		else
			Object.TurnDirection = true
		end if
	else
		if Object.BumperOscillation > -0x300000		// -0x380000
			Object.BumperOscillation -= 0x10000		// 0x10000
		else
			Object.TurnDirection = false
		end if
	end if

	if HUD.TimePeriod != PROPVAL_PAST
		RotatingDrum_Timer++
		RotatingDrum_Timer %= 20
	end if
end sub


sub ObjectPlayerInteraction
	TempValue2 = Player.XPos
	TempValue2 >>= 16
	
	TempValue4 = TempValue2
	TempValue2 -= 8
	TempValue4 += 8

	TempValue3 = Player.YPos
	TempValue3 >>= 16

	if Player.YVelocity >= 0
		TempValue3 += Player.CollisionBottom
		TempValue3 -= 2
	else
		TempValue3 += Player.CollisionTop
		TempValue3 += 2
	end if

	TempValue5 = TempValue3
	TempValue7 = 0
	Get16x16TileInfo(TempValue0, TempValue2, TempValue3, 6)
	switch TempValue0
	case 1
		TempValue7 = 1
		break

	case 3
		TempValue7 = 3
		Get16x16TileInfo(TempValue6, TempValue2, TempValue3, 1)
		break

	case 5
		TempValue7 = 2
		break

	end switch

	Get16x16TileInfo(TempValue0, TempValue4, TempValue5, 6)
	switch TempValue0
	case 1
		TempValue7 = 1
		break

	case 3
		TempValue7 = 3
		Get16x16TileInfo(TempValue6, TempValue4, TempValue5, 1)
		break

	case 5
		TempValue7 = 2
		break

	end switch

	switch TempValue7
	case 1
		if Player.YVelocity >= 0
			Player.Gravity = GRAVITY_AIR
			PlayStageSfx(1, 0)
				
#platform: Use_Standalone
	#platform: Use_Haptics
			HapticEffect(11, 0, 0, 0)
	#endplatform
#endplatform

			if Player.State == PlayerObject_HandleTailsFlight
				Player.YVelocity = -0x40000	// -0x40000
			else
				Player.YVelocity = -458752	// -0x70000
			end if

			if Player.State == PlayerObject_Knockback
				
#platform: Use_Origins
				Player.State = PlayerObject_ResetDropDash
#endplatform

#platform: Use_Standalone
				Player.State = PlayerObject_HandleAir
#endplatform
				Player.Timer = 0
			end if
		else
			if Player.Gravity == GRAVITY_AIR
				Player.YVelocity = 0x40000	// 0x40000
				PlayStageSfx(1, 0)
			end if
		end if
		break

	case 2
		if Player.Gravity == GRAVITY_AIR
			PlayStageSfx(6, 0)
				
#platform: Use_Standalone
	#platform: Use_Haptics
			HapticEffect(11, 0, 0, 0)
	#endplatform
#endplatform

			if Player.YVelocity >= 0
				Player.YVelocity = -0x40000	// -0x40000
			else
				Player.YVelocity = 0x40000	// 0x40000
			end if
		end if
		break

	case 3
		if Player.Gravity == GRAVITY_AIR
			PlayStageSfx(1, 0)
				
#platform: Use_Standalone
	#platform: Use_Haptics
			HapticEffect(11, 0, 0, 0)
	#endplatform
#endplatform

			if Player.YVelocity >= 0
				if Player.State == PlayerObject_HandleTailsFlight
					Player.YVelocity = -0x40000	// -0x40000
				else
					Player.YVelocity = -458752	// -0x70000
				end if
			else
				Player.YVelocity = 0x40000	// 0x40000
			end if

			if TempValue6 == 0
				if Player.XVelocity > 0
					FlipSign(Player.XVelocity)
					Player.XVelocity *= 5
					Player.XVelocity >>= 2
					if Player.XVelocity > -0x20000	// -0x20000
						Player.XVelocity = -0x20000	// -0x20000
					end if
				else
					if Player.XVelocity > -0x20000	// -0x20000
						Player.XVelocity = -0x20000	// -0x20000
					end if
				end if
			else
				if Player.XVelocity < 0
					FlipSign(Player.XVelocity)
					Player.XVelocity *= 5
					Player.XVelocity >>= 2
					if Player.XVelocity < 0x20000	// 0x20000
						Player.XVelocity = 0x20000	// 0x20000
					end if
				else
					if Player.XVelocity < 0x20000	// 0x20000
						Player.XVelocity = 0x20000	// 0x20000
					end if
				end if
			end if
			Player.Speed = Player.XVelocity
		end if
		break

	end switch
	TempValue3 = Player.YPos
	TempValue3 >>= 16

	TempValue5 = TempValue3
	TempValue3 -= 6
	TempValue5 += 6

	TempValue2 = Player.XPos
	TempValue2 >>= 16

	if Player.XVelocity > 0
		TempValue2 += 8
	else
		TempValue2 -= 8
	end if

	TempValue4 = TempValue2
	TempValue7 = 0
	Get16x16TileInfo(TempValue0, TempValue2, TempValue3, 6)
	switch TempValue0
	case 3
		TempValue7 = 1
		break

	case 5
		TempValue7 = 2
		break

	case 7
		TempValue7 = 1
		break

	end switch

	Get16x16TileInfo(TempValue0, TempValue4, TempValue5, 6)
	switch TempValue0
	case 3
		TempValue7 = 1
		break

	case 5
		TempValue7 = 2
		break

	case 7
		TempValue7 = 1
		break

	end switch

	switch TempValue7
	case 1
		if Player.XVelocity > 0
			Player.XVelocity = -0x50000	// -0x50000
		else
			Player.XVelocity = 0x50000	// 0x50000
		end if
		Player.Speed = Player.XVelocity
		PlayStageSfx(1, 0)
		break

	case 2
		if Player.XVelocity > 0
			Player.XVelocity = -0x40000	// -0x40000
		else
			Player.XVelocity = 0x40000	// 0x40000
		end if
		Player.Speed = Player.XVelocity
		PlayStageSfx(6, 0)
				
#platform: Use_Standalone
	#platform: Use_Haptics
		HapticEffect(11, 0, 0, 0)
	#endplatform
#endplatform
		break

	end switch
end sub


sub ObjectDraw
	Object.DeformTimer++
	if Object.DeformTimer > 3
		TileLayer[0].DeformationOffsetW++
		TileLayer[1].DeformationOffset++
		Object.DeformTimer = 0
	end if
end sub


sub ObjectStartup
	Object[29].Type = TypeName[R3Setup]
	Object[29].Priority = 1
	Object[29].DrawOrder = 0
	SetLayerDeformation(2, 128, 6, 0, 0, 0)
	SetLayerDeformation(1, 64, 4, 0, 0, 0)
	Stage.WaterLevel = 1472
	if HUD.TimePeriod == PROPVAL_PAST
		RotatingDrum_Timer = 1
	else
		RotatingDrum_Timer = 0
	end if
end sub
