	TempValue0 %= 10

	if TempValue0 == 0
		Object.Timer++
		if Object.Timer > 3
			Object.Timer = 0

			Object.State++
			Object.State %= 18
		end if
	else
		Object.PlayerFloor.Y = 0
	end if

	Object.Priority = PRIORITY_ACTIVE
	
	TempValue0 = Object.iXPos

	end if

sub ObjectPlayerInteraction
	switch Object.State
	case ROTATINGDRUM_FRAME_2
		PlayerObjectCollision(C_PLATFORM, -24, -8, 24, 12)
		if CheckResult == true
			Player.YPos += Object.PlayerFloor.Y
		end if
		break

	case ROTATINGDRUM_FRAME_3
	case ROTATINGDRUM_FRAME_4
	case ROTATINGDRUM_FRAME_5
	case ROTATINGDRUM_FRAME_6
		PlayerObjectCollision(C_PLATFORM, -24, -4, 24, 12)
		if CheckResult == true
			Player.YPos += Object.PlayerFloor.Y
		end if
		break
		
	end switch
end sub


sub ObjectDraw
	Object.YPos += Object.PlayerFloor.Y

end sub
