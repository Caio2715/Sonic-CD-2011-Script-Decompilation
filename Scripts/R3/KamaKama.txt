sub ObjectMain
	Object.Value1++
	Object.Value1 &= 15
	switch Object.State
	case 0
		ObjectTileGrip(0, 0, 19, 0)
		if Object.XPos > Player.XPos
			Object.Direction = FACING_RIGHT
		else
			Object.Direction = FACING_LEFT
		end if
		PlayerObjectCollision(C_TOUCH, -96, -48, 96, 48)
		if CheckResult == 1
			Object.State = 1
			Object.Priority = 1
			if Object.PropertyValue == 0
				CreateTempObject(TypeName[KamaKamaBlade], 24, Object.XPos, Object.YPos)
				Object[TempObjectPos].Direction = Object.Direction
				Object[TempObjectPos].iYPos -= 6
				Object[TempObjectPos].Value2 = Object.EntityNo
				if Object.Direction == FACING_RIGHT
					Object[TempObjectPos].iXPos -= 8
				else
					Object[TempObjectPos].iXPos += 8
				end if
				CreateTempObject(TypeName[KamaKamaBlade], 56, Object.XPos, Object.YPos)
				Object[TempObjectPos].Direction = Object.Direction
				Object[TempObjectPos].iYPos -= 4
				Object[TempObjectPos].Value2 = Object.EntityNo
				if Object.Direction == FACING_RIGHT
					Object[TempObjectPos].iXPos -= 16
				else
					Object[TempObjectPos].iXPos += 16
				end if
			end if
		end if
		break
	case 1
		if Object.Value0 < 110
			Object.Value0++
		else
			Object.State = 2
			Object.Value0 = 0
		end if
		break
	case 2
		if Object.Value0 < 48
			Object.Value0++
		else
			Object.State = 3
			Object.Value0 = 0
			Object.Frame = 0
			if Object.Direction == FACING_RIGHT
				Object.Value2 = -65536
			else
				Object.Value2 = 65536
			end if
			if Object.PropertyValue > 0
				Object.Value2 >>= 1
			end if
		end if
		break
	case 3
		Object.Frame++
		Object.Frame &= 31
		Object.XPos += Object.Value2
		ObjectTileGrip(0, 0, 19, 0)
		if CheckResult == 0
			Object.State = 4
			Object.Value0 = 60
		end if
		break
	case 4
		Object.Frame++
		Object.Frame &= 31
		if Object.Value0 > 0
			Object.Value0--
		else
			Object.State = 3
			FlipSign(Object.Value2)
			if Object.Direction == FACING_RIGHT
				Object.Direction = FACING_LEFT
			else
				Object.Direction = FACING_RIGHT
			end if
		end if
		break
	end switch
	if Object.OutOfBounds == 1
		Object.Priority = 0
		Object.State = 0
		Object.Value0 = 0
		Object.Value1 = 0
		Object.Frame = 0
		Object.XPos = Object.Value3
		Object.YPos = Object.Value4
	end if
	CallFunction(StageSetup_CheckGoodFuture)
end sub


sub ObjectPlayerInteraction
	PlayerObjectCollision(C_TOUCH, -16, -16, 16, 20)
	if CheckResult == 1
		CallFunction(PlayerObject_BadnikBreak)
	end if
end sub
