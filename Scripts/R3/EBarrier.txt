sub ObjectMain
	switch Object.State
	case 1
		if Object.YPos > 84934656
			Object.YPos -= 163840
		else
			Object.YPos = 84934656
			Object.State = 2
#platform: Use_Haptics
			HapticEffect(76, 0, 0, 0)
#endplatform
		end if
		Object.Frame++
		Object.Frame %= 9
		break
	case 2
		TempValue0 = Object.Value0
		TempValue0 &= 3
		if TempValue0 == 0
			Rand(TempValue0, 96)
			TempValue0 -= 48
			TempValue0 <<= 16
			TempValue0 += Object.XPos
			Rand(TempValue1, 40)
			TempValue1 -= 20
			TempValue1 <<= 16
			TempValue1 += Object.YPos
			CreateTempObject(TypeName[Explosion], 0, TempValue0, TempValue1)
			PlaySfx(22, 0)
		end if
		Object.Value0++
		if Object.Value0 == 120
			Object.Type = TypeName[BlankObject]
		end if
		break
	end switch
end sub
