// Aliases
#alias Object.Value0 : Object.Speed


sub ObjectDraw
	Object.Speed += 8
	Object.Speed &= 2047
	TempValue0 = Object.Speed
	FlipSign(TempValue0)
	MatrixTranslateXYZ(MAT_WORLD, -512, 720, TempValue0)
	Draw3DScene()
end sub


sub ObjectStartup
	LoadSpriteSheet("Title/Clouds.gif")
	3DScene.NoVertices = 0
	3DScene.NoFaces = 0
	ArrayPos0 = 3DScene.NoVertices
	ArrayPos1 = 3DScene.NoFaces
	MatrixTranslateXYZ(MAT_WORLD, 0, 0, 0)
	MatrixRotateXYZ(MAT_WORLD, 0, 0, 0)
	MatrixRotateXYZ(MAT_VIEW, 8, 0, 0)
	ArrayPos0 = 0
	ArrayPos1 = 0
	TempValue2 = 0	//TempValue2 is the starting point of the clouds after loop
	TempValue3 = 0	//TempValue3 is the minimal value for the amount of clouds to generate
	TempValue4 = 0	//TempValue4 is the width of the cloud sprite
	TempValue5 = 0	//TempValue5 is the height of the cloud sprite
	while TempValue3 < 10	//Makes 10 Cloud "Rows"
		TempValue0 = 0
		TempValue1 = -8192
		TempValue4 = 0
		while TempValue0 < 16	//Long Story Short, sets the angles for 16 cloud "Columns"
			VertexBuffer[ArrayPos0].x = TempValue1
			VertexBuffer[ArrayPos0].y = 0
			VertexBuffer[ArrayPos0].z = TempValue2
			VertexBuffer[ArrayPos0].u = TempValue4
			VertexBuffer[ArrayPos0].v = TempValue5
			ArrayPos0++
			TempValue2 += 512
			VertexBuffer[ArrayPos0].x = TempValue1
			VertexBuffer[ArrayPos0].y = 0
			VertexBuffer[ArrayPos0].z = TempValue2
			VertexBuffer[ArrayPos0].u = TempValue4
			VertexBuffer[ArrayPos0].v = TempValue5
			VertexBuffer[ArrayPos0].v += 63
			ArrayPos0++
			TempValue2 -= 512
			TempValue1 += 1024
			VertexBuffer[ArrayPos0].x = TempValue1
			VertexBuffer[ArrayPos0].y = 0
			VertexBuffer[ArrayPos0].z = TempValue2
			VertexBuffer[ArrayPos0].u = TempValue4
			VertexBuffer[ArrayPos0].u += 63
			VertexBuffer[ArrayPos0].v = TempValue5
			ArrayPos0++
			TempValue2 += 512
			VertexBuffer[ArrayPos0].x = TempValue1
			VertexBuffer[ArrayPos0].y = 0
			VertexBuffer[ArrayPos0].z = TempValue2
			VertexBuffer[ArrayPos0].u = TempValue4
			VertexBuffer[ArrayPos0].u += 63
			VertexBuffer[ArrayPos0].v = TempValue5
			VertexBuffer[ArrayPos0].v += 63
			TempValue4 += 64
			TempValue4 &= 255
			TempValue2 -= 512
			ArrayPos0 -= 3
			FaceBuffer[ArrayPos1].a = ArrayPos0
			FaceBuffer[ArrayPos1].b = ArrayPos0
			FaceBuffer[ArrayPos1].b += 2
			FaceBuffer[ArrayPos1].c = ArrayPos0
			FaceBuffer[ArrayPos1].c += 1
			FaceBuffer[ArrayPos1].d = ArrayPos0
			FaceBuffer[ArrayPos1].d += 3
			FaceBuffer[ArrayPos1].Flag = 0 // The FACE_TEXTURED_3D flag, since the clouds sprites are gonna be projected onto the face
			ArrayPos0 += 4
			ArrayPos1++
			TempValue0++
			3DScene.NoVertices += 4
			3DScene.NoFaces++
		loop
		TempValue5 += 64
		TempValue5 &= 255
		TempValue2 += 512
		TempValue3++
	loop
end sub
