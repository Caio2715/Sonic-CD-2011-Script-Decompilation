	case 1
		if Object.Value2>0
			Object.Value2--
		else
			PlayStageSfx(3,0)
			Object.State=4
		endif
		break
	case 2
		if Object.Value2>0
			Object.Value2--
		else
			PlayStageSfx(3,0)
			Object.State=3
		endif
		break
	case 3
		Object.Value2++
		if Object.Value2==12
			Object.Value2=0
			if Object.Value0<Object.Value1
				Object.Value3--
				Object.Value0+=16
				TempValue0=Object.Value0
				TempValue0-=8
				TempValue0<<=16
				TempValue0+=Object.XPos
				TempValue1=Object.YPos
				TempValue1-=524288
				CreateTempObject(TypeName[CPlatform],0,TempValue0,TempValue1)
				Object[TempObjectPos].State=6
				Object[TempObjectPos].Value4=-65536
				TempValue1+=1048576
				CreateTempObject(TypeName[CPlatform],0,TempValue0,TempValue1)
				Object[TempObjectPos].State=6
			else
				ResetObjectEntity(Object.EntityNo,TypeName[BlankObject],0,0,0)
			endif
		endif
		break
	case 4
		Object.Value2++
		if Object.Value2==12
			Object.Value2=0
			if Object.Value1>Object.Value0
				Object.Value3--
				Object.Value1-=16
				TempValue0=Object.Value1
				TempValue0+=8
				TempValue0<<=16
				TempValue0+=Object.XPos
				TempValue1=Object.YPos
				TempValue1-=524288
				CreateTempObject(TypeName[CPlatform],0,TempValue0,TempValue1)
				Object[TempObjectPos].State=6
				Object[TempObjectPos].Value4=-65536
				TempValue1+=1048576
				CreateTempObject(TypeName[CPlatform],0,TempValue0,TempValue1)
				Object[TempObjectPos].State=6
			else
				ResetObjectEntity(Object.EntityNo,TypeName[BlankObject],0,0,0)
			endif
		endif
		break
	case 5
		if Object.OutOfBounds==1
			Object.State=0
			Object.Value3=Object.PropertyValue
			Object.Value0=Object.PropertyValue
			Object.Value0<<=3
			Object.Value1=Object.Value0
			FlipSign(Object.Value0)
			Object.Priority=0
			Object.Value2=0
		endif
		break
	case 6
		Object.Value4+=16384
		if Object.Value4>0
			Object.YPos+=Object.Value4
		endif
		if Object.OutOfBounds==1
			Object.Type=TypeName[BlankObject]
		endif
		break
	endswitch
endsub


sub ObjectPlayerInteraction
	if Object.State==0
		PlayerObjectCollision(C_PLATFORM,Object.Value0,-16,Object.Value1,16)
		if CheckResult==1
			Object.Value2=1
			Object.Priority=1
			if Player.XPos>Object.XPos
				Object.State=1
			else
				Object.State=2
			endif
		endif
	else
		PlayerObjectCollision(C_PLATFORM,Object.Value0,-16,Object.Value1,16)
	endif
endsub
