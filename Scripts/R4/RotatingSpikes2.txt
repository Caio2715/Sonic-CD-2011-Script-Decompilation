	if Object.PropertyValue < 128
		Object.Value0 += 2
		Object.Value0 &= 511
	else
		Object.Value0 -= 2

			Object.Value0 += 512
		end if
	end if
end sub


sub ObjectPlayerInteraction
	TempValue0 = Object.XPos
	TempValue1 = Object.YPos
	TempValue2 = Object.Value0
	TempValue3 = 0
	if Object.PropertyValue < 128
		while TempValue3 < 4
			Cos(Object.XPos, TempValue2)
			Object.XPos <<= 13
			Object.XPos += TempValue0
			Sin(Object.YPos, TempValue2)
			Object.YPos <<= 13
			Object.YPos += TempValue1
			PlayerObjectCollision(C_TOUCH, -12, -12, 12, 12)
			if CheckResult == 1
				CallFunction(PlayerObject_Hit)
			end if
			TempValue2 += 16

		loop
	else
		TempValue2 += 64
		while TempValue3 < 4
			Cos(Object.XPos, TempValue2)
			Object.XPos <<= 13
			Object.XPos += TempValue0
			Sin(Object.YPos, TempValue2)
			Object.YPos <<= 13
			Object.YPos += TempValue1
			PlayerObjectCollision(C_TOUCH, -12, -12, 12, 12)
			if CheckResult == 1
				CallFunction(PlayerObject_Hit)
			end if
			TempValue2 -= 16
			TempValue3++
		loop
	end if
	Object.XPos = TempValue0
	Object.YPos = TempValue1
end sub
