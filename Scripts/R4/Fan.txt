sub ObjectMain
	switch Object.State
	case 0
	case 2
		if Object[-1].PropertyValue==1
			Object.PropertyValue=1
			Object.State++
		endif
		break
	case 1
	case 3
		if Object[24].PropertyValue<2
			Object.Frame++
			if Object.Frame>9
				Object.Frame=0
			endif
		else
			Object.Frame=2
		endif
		break
	endswitch
endsub


sub ObjectDraw
	switch Object.State
	case 0
		DrawSprite(0)
		break
	case 1
		TempValue0=Object.Frame
		TempValue0>>=1
		DrawSprite(TempValue0)
		break
	case 2
		DrawSpriteFX(5,FX_FLIP,Object.XPos,Object.YPos)
		break
	case 3
		TempValue0=Object.Frame
		TempValue0>>=1
		TempValue0+=5
		DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		break
	endswitch
endsub
