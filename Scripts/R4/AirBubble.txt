sub ObjectMain
	if Object.State == 0
		Object.Value0++
		if Object.Value0 > 15
			Object.Value0 = 0
			if Object.Frame == 6
				Object.Type = TypeName[BlankObject]
			end if
			if Object.Frame < Object.PropertyValue
				Object.Frame++
			end if
		end if
		Object.YPos += Object.Value3
		if Player.Animation == ANI_FANROTATE
			if Object.PropertyValue < 3
				Object.Value2 += 262144
			end if
		end if
		if Player.Animation == ANI_CLINGING
			if Object.PropertyValue < 3
				Object.Value2 += 262144
			end if
		end if
		if Object.Frame < 6
			Sin(Object.XPos, Object.Value1)
			Object.XPos <<= 9
			Object.XPos += Object.Value2
			Object.Value1 += 4
			Object.Value1 &= 511
		end if
		if Object.iYPos < Stage.WaterLevel
			if Object.PropertyValue == 5
				Object.Frame = 6
				Object.PropertyValue = 6
				Object.Value0 = 0
				Object.Value3 = 0
			else
				if Object.PropertyValue < 5
					Object.Type = TypeName[BlankObject]
				end if
			end if
		end if
	else
		if Object.Value0 < 20
			Object.Value0++
			Player.Animation = ANI_BREATHING
		else
			Object.Type = TypeName[BlankObject]
			Player.Animation = ANI_WALKING
			Player.AnimationSpeed = 20
		end if
	end if
	if Object.OutOfBounds == 1
		Object.Type = TypeName[BlankObject]
	end if
end sub


sub ObjectPlayerInteraction
	if Object.Frame == 5
		PlayerObjectCollision(C_TOUCH, -4, -4, 4, 4)
		if CheckResult == 1
			Object.Frame = 6
			Object.State = 1
			Object.Value0 = 0
			Object.Value3 = 0
			Player.YVelocity = 0
			Player.XVelocity = 0
			Player.Speed = 0
			Player.Animation = ANI_BREATHING
			Player.AirTimer = 0
			PlayStageSfx(6, 0)
		end if
	end if
end sub

	DrawSprite(Object.Frame)
