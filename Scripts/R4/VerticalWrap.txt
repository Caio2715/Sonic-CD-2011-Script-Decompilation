// Aliases
#alias Object.Value0	:	Object.WaterEntityPos
#alias Object.Value1	:	Object.MinWaterHeight
#alias Object.Value2	:	Object.ExitWrap	// Wild guess based on what happens when this is true

#alias 0	:	PRIORITY_ACTIVE_BOUNDS
#alias 1	:	PRIORITY_ACTIVE


sub ObjectMain
	if Player.XPos > Object.XPos
		Stage.YBoundary2 = 2048
		if Object.OutOfBounds == true
			Object.Priority = PRIORITY_ACTIVE_BOUNDS
		end if

		if Stage.NewWaterLevel > Object.MinWaterHeight
			Stage.NewWaterLevel = Object.MinWaterHeight
			ArrayPos0 = Object.WaterEntityPos
			Object[ArrayPos0].YPos = Screen.YOffset
			Object[ArrayPos0].YPos += 256
			Object[ArrayPos0].YPos <<= 16
			if Object.ExitWrap == 1
				// this gets triggered once you leave the wrap section
				if Object[ArrayPos0].YPos < Object.MinWaterHeight
					Object[ArrayPos0].YPos = Object.MinWaterHeight
				end if
			end if
		end if
	else
		if Object.PropertyValue == 0
			// 0xC000000, Y - 3072 in map coords
			if Stage.NewWaterLevel < 201326592
				Stage.NewWaterLevel = 201326592
				ArrayPos0 = Object.WaterEntityPos
				Object[ArrayPos0].YPos = 201326592
			end if
		else
			if Player.YPos > Object.YPos
				// 0xC000000, Y - 3072 in map coords
				if Stage.NewWaterLevel < 201326592
					Stage.NewWaterLevel = 201326592
					ArrayPos0 = Object.WaterEntityPos
					Object[ArrayPos0].YPos = 201326592
				end if
			end if
		end if
		
		Stage.YBoundary2 = 2304
		Object.Priority = PRIORITY_ACTIVE
		if Player.YVelocity >= 0
			if Screen.YOffset > 2047
				Player.YPos -= 134217728	// 0x8000000, Y - 2048 in map coords
				Screen.YOffset -= 2048
			end if
		end if
		if Player.YVelocity <= 0
			if Screen.YOffset < 1
				Player.YPos += 134217728	// 0x8000000, Y - 2048 in map coords
				Screen.YOffset += 2048
			end if
		end if
	end if
end sub


sub ObjectStartup
	ArrayPos0 = 32
	while ArrayPos0 < 1056
		if Object[ArrayPos0].Type == TypeName[Water]
			TempValue0 = ArrayPos0
		end if
		ArrayPos0++
	loop

	ArrayPos0 = 32
	while ArrayPos0 < 1056
		if Object[ArrayPos0].Type == TypeName[VerticalWrap]
			Object[ArrayPos0].WaterEntityPos = TempValue0
			switch Object[ArrayPos0].PropertyValue
			case 0
			case 1
				Object[ArrayPos0].MinWaterHeight = 87293952
				Object[ArrayPos0].ExitWrap = 0
				break
			case 2
			case 3
				Object[ArrayPos0].PropertyValue -= 2
				Object[ArrayPos0].MinWaterHeight = 94371840
				Object[ArrayPos0].ExitWrap = 1
				break
			case 4
			case 5
				Object[ArrayPos0].PropertyValue -= 2
				Object[ArrayPos0].MinWaterHeight = 0x5100000
				Object[ArrayPos0].ExitWrap = 0
				break
			end switch
		end if
		ArrayPos0++
	loop

	if Player.XPos > 52428800
		Stage.YBoundary2 = 2048
	end if
end sub
