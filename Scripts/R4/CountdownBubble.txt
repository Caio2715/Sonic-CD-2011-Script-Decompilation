sub ObjectMain
	switch Object.State
	case 0
		Object.YPos += Object.Value3
		Object.Value0++
		if Object.Value0 > 6
			Object.Value0 = 0
			Object.Frame++
		end if
		if Object.Frame < 4
			Sin(Object.XPos, Object.Value1)
			Object.XPos <<= 9
			Object.XPos += Object.Value2
			Object.Value1 += 4
			Object.Value1 &= 511
		else
			TempValue0 = Object.PropertyValue
			TempValue0 *= 3
			Object.Frame += TempValue0
			Object.Value2 = Object.iXPos
			Object.Value2 -= Screen.XOffset
			Object.Value4 = Object.iYPos
			Object.Value4 -= Screen.YOffset
			Object.State++
		end if
		break
	case 1
		Object.Value0++
		if Object.Value0 == 6
			Object.Value0 = 0
			Object.Frame++
			Object.State++
		end if
		break
	case 2
		Object.Value0++
		if Object.Value0 == 6
			Object.Value0 = 0
			Object.Frame++
			Object.State++
		end if
		break
	case 3
		Object.Value0++
		if Object.Value0 == 8
			Object.Value0 = 0
			Object.Frame--
			Object.State++
		end if
		break
	case 4
		Object.Value0++
		if Object.Value0 == 8
			Object.Value0 = 0
			Object.Frame++
			Object.State++
		end if
		break
	case 5
		Object.Value0++
		if Object.Value0 == 8
			Object.Value0 = 0
			Object.Frame--
			Object.State++
		end if
		break
	case 6
		Object.Value0++
		if Object.Value0 == 8
			Object.Value0 = 0
			Object.Frame++
			Object.State++
		end if
		break
	case 7
		Object.Value0++
		if Object.Value0 == 8
			Object.Value0 = 0
			Object.Frame--
			Object.State++
		end if
		break
	case 8
		Object.Value0++
		if Object.Value0 == 9
			Object.Type = TypeName[BlankObject]
		end if
		break
	end switch
end sub


sub ObjectDraw
	if Object.State == 0
		DrawSprite(Object.Frame)
	else
		DrawSpriteScreenXY(Object.Frame, Object.Value2, Object.Value4)
	end if
end sub
