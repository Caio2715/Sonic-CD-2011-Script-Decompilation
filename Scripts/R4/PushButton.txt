// Aliases
#alias 0	:	PUSHBUTTON_INACTIVE
#alias 1	:	PUSHBUTTON_PRESSED


sub ObjectPlayerInteraction
	if Object.PropertyValue == PUSHBUTTON_INACTIVE
		if Player.YVelocity >= 0
			PlayerObjectCollision(C_BOX, -16, -8, 16, 8)
			if CheckResult == true
				Object.PropertyValue = PUSHBUTTON_PRESSED
				Player.YPos += 524288	// 0x80000
				PlayStageSfx(8, 0)
			else
				Object.PropertyValue = PUSHBUTTON_INACTIVE
			end if
		else
			Object.PropertyValue = PUSHBUTTON_INACTIVE
		end if
	else
		PlayerObjectCollision(C_TOUCH, -16, -2, 16, 8)
		if CheckResult == true
			if Player.YVelocity > -1
				Object.PropertyValue = PUSHBUTTON_PRESSED
				Player.Gravity = 0
				Player.YVelocity = 0

				Player.YPos = Player.CollisionBottom
				FlipSign(Player.YPos)
				Player.YPos <<= 16
				Player.YPos += Object.YPos
			end if
		else
			Object.PropertyValue = PUSHBUTTON_INACTIVE
			if Player.YVelocity > -1
				Player.Gravity = 0
				Player.YVelocity = 0

				Player.YPos = Player.CollisionBottom
				FlipSign(Player.YPos)
				Player.YPos -= 8
				Player.YPos <<= 16
				Player.YPos += Object.YPos
			end if
		end if
	end if
end sub


sub ObjectDraw
	DrawSprite(Object.PropertyValue)
end sub


sub ObjectStartup
	LoadSpriteSheet("R4/Objects.gif")
	SpriteFrame(-16, -8, 32, 16, 130, 35)	// #0 - Push Button
	SpriteFrame(-16, -8, 32, 16, 130, 52)	// #1 - Push Button pressed
end sub

// ========================
// Editor Subs
// ========================

sub RSDKDraw
	DrawSprite(0)
end sub

sub RSDKLoad
	LoadSpriteSheet("Global/Display.gif")
	SpriteFrame(-16, -16, 32, 32, 1, 143)		// #0 - "Script" Icon

	SetVariableAlias(ALIAS_VAR_PROPVAL, "unused")
end sub
