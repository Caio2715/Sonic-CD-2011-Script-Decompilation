#alias 48: TYPE_BOSSBUBBLE2


sub ObjectMain
	switch Object.State
	case 0
		if Object.Value0<20
			Object.Value0++
		else
			Object.State++
		endif
		break
	case 1
		Object.State++
		break
	case 3
		if Object.Value0<22
			Object.Value0++
		else
			Object.State++
		endif
		break
	case 4
		if Object.Value0<42
			Object.Value0++
		else
			Object.Type=TypeName[BlankObject]
		endif
		break
	endswitch
endsub


sub ObjectPlayerInteraction
	if Object.State==2
		if Player.Animation==ANI_JUMPING
			PlayerObjectCollision(C_TOUCH,-8,-8,8,8)
			if CheckResult==1
				Object.State=3
				FlipSign(Player.Speed)
				FlipSign(Player.XVelocity)
				Player.YVelocity=262144
				Player.Animation=ANI_BREATHING
				Player.AirTimer=0
				PlayStageSfx(6,0)
#platform: Use_Haptics
				HapticEffect(19,0,0,0)
#endplatform
			endif
		endif
		if Player.Animation==ANI_FLYING
			PlayerObjectCollision(C_TOUCH,-8,-8,8,8)
			if CheckResult==1
				Object.State=3
				FlipSign(Player.Speed)
				FlipSign(Player.XVelocity)
				Player.YVelocity=262144
				Player.Animation=ANI_BREATHING
				Player.State=PlayerObject_HandleAir
				Player.AirTimer=0
				PlayStageSfx(6,0)
#platform: Use_Haptics
				HapticEffect(19,0,0,0)
#endplatform
			endif
		endif
	endif
endsub


sub ObjectDraw
	DrawSprite(Object.State)
endsub
