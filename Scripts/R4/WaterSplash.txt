sub ObjectMain
	Object.iYPos = Stage.WaterLevel
	Object.Value0++
	if Object.Value0 > 4
		Object.Value0 = 0
		Object.Frame++

			Object.Type = TypeName[BlankObject]
		end if
	end if
end sub
