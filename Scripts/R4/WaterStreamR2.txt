// Aliases
#alias Object.Value1	:	Object.HitboxLeft
#alias Object.Value2	:	Object.HitboxRight


sub ObjectPlayerInteraction
	if Object[-1].PropertyValue == 1
		Object.PropertyValue = 1
		PlayerObjectCollision(C_TOUCH, Object.HitboxLeft, -32, Object.HitboxRight, 32)
		if CheckResult == true
			if Player.State != PlayerObject_Knockback
				if Player.State != PlayerObject_Death
					if Player.State != PlayerObject_Drown
						if Player.State != PlayerObject_Blank
							Player.WaterStream = true
							Player.State = PlayerObject_DebugMode
							Player.Animation = ANI_FANROTATE
							Player.XVelocity = 262144	// 0x40000
							Player.Speed = 262144		// 0x40000
							Player.YVelocity = 0
						end if
					end if
				end if
			end if
		else
			if Player.State == PlayerObject_DebugMode
				if Player.WaterStream == false
					Player.State = PlayerObject_HandleAir
					Player.ControlMode = 0
					Player.Animation = ANI_WALKING
				end if
			end if
		end if
	end if
end sub


sub ObjectStartup
	ArrayPos0 = 32
	while ArrayPos0 < 1056
		if Object[ArrayPos0].Type == TypeName[WaterStreamR2]
			Object[ArrayPos0].HitboxLeft = Object[ArrayPos0].PropertyValue
			Object[ArrayPos0].HitboxLeft <<= 3
			Object[ArrayPos0].HitboxRight = Object[ArrayPos0].HitboxLeft
			FlipSign(Object[ArrayPos0].HitboxLeft)
			Object[ArrayPos0].PropertyValue = 0
		end if
		ArrayPos0++
	loop
end sub
