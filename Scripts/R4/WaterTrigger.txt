sub ObjectPlayerInteraction
	switch Object.PropertyValue
	case 0
		if Object[-1].PropertyValue == 1
			PlayerObjectCollision(C_TOUCH, -112, -64, 0, 160)
			if CheckResult == true
				if Player.State != Player_State_Death
					if Player.State != Player_State_Drown
						Player.State = Player_State_WaterCurrent
						Player.Animation = ANI_FANROTATE
						
						Player.ControlMode = -1
						Player.Left = false
						Player.Right = false
						Player.Up = false
						Player.Down = false

						if Player.YPos < Object.YPos
							Player.XVelocity = -0x40000	// -0x40000
							Player.Speed = -0x40000		// -0x40000
						else
							Player.XVelocity = 0
							Player.Speed = 0
						end if
						Player.YVelocity = -0x40000	// -0x40000
					end if
				end if
			else
				if Player.State == Player_State_WaterCurrent

#platform: Use_Origins
				Player.State = Player_State_Air_NoDropDash
#endplatform

#platform: Use_Standalone
				Player.State = Player_State_Air
#endplatform
					Player.ControlMode = 0
					Player.Animation = ANI_WALKING
				end if
			end if
		end if
		break
		
	case 1
		PlayerObjectCollision(C_TOUCH, -128, 0, 112, 64)
		if CheckResult == true
			if Player.State != Player_State_Death
				if Player.State != Player_State_Drown
					Player.State = Player_State_WaterCurrent
					Player.Animation = ANI_FANROTATE

					Player.ControlMode = -1
					Player.Left = false
					Player.Right = false
					Player.Up = false
					Player.Down = false

					Player.XVelocity = 0x60000
					Player.Speed = 0x60000	
					Player.YVelocity = 0x20000	// 0x20000
				end if
			end if
		else
			if Player.State == Player_State_WaterCurrent
				TempValue0 = Object.XPos
				TempValue0 += 8650752	// 0x840000
				if Player.XPos > TempValue0

#platform: Use_Origins
				Player.State = Player_State_Air_NoDropDash
#endplatform

#platform: Use_Standalone
				Player.State = Player_State_Air
#endplatform
					Player.ControlMode = 0
					Player.Animation = ANI_WALKING
				end if
			end if
		end if
		break

	case 2
		PlayerObjectCollision(C_TOUCH, -128, 0, 112, 64)
		if CheckResult == true
			if Player.State != Player_State_Death
				if Player.State != Player_State_Drown
					Player.State = Player_State_WaterCurrent
					Player.Animation = ANI_FANROTATE

					Player.ControlMode = -1
					Player.Left = false
					Player.Right = false
					Player.Up = false
					Player.Down = false

					Player.XVelocity = -0x60000
					Player.Speed = -0x60000	
					Player.YVelocity = 0x20000	//  0x20000
				end if
			end if
		else
			if Player.State == Player_State_WaterCurrent
				TempValue0 = Object.XPos
				TempValue0 += 8650752	// 0x840000
				if Player.XPos > TempValue0

#platform: Use_Origins
				Player.State = Player_State_Air_NoDropDash
#endplatform

#platform: Use_Standalone
				Player.State = Player_State_Air
#endplatform
					Player.ControlMode = 0
					Player.Animation = ANI_WALKING
				end if
			end if
		end if
		break

	case 3
		PlayerObjectCollision(C_TOUCH, -128, 0, 112, 64)
		if CheckResult == true
			if Player.State != Player_State_Death
				if Player.State != Player_State_Drown
					Player.State = Player_State_WaterCurrent
					Player.Animation = ANI_FANROTATE

					Player.ControlMode = -1
					Player.Left = false
					Player.Right = false
					Player.Up = false
					Player.Down = false

					Player.XVelocity = -0x60000
					Player.Speed = -0x60000	
					Player.YVelocity = 0x20000	//  0x20000
				end if
			end if
		else
			if Player.YPos > Object.YPos
				if Player.State == Player_State_WaterCurrent

#platform: Use_Origins
				Player.State = Player_State_Air_NoDropDash
#endplatform

#platform: Use_Standalone
				Player.State = Player_State_Air
#endplatform
					Player.ControlMode = 0
					Player.Animation = ANI_WALKING
				end if
			end if
		end if
		break

	case 4
		PlayerObjectCollision(C_TOUCH, 0, -64, 256, 64)
		if CheckResult == true
			if Player.State != Player_State_Death
				if Player.State != Player_State_Drown
					Player.State = Player_State_WaterCurrent
					Player.Animation = ANI_FANROTATE

					Player.XVelocity = 0x40000	// 0x40000
					Player.Speed = 0x40000		// 0x40000
					Player.YVelocity = 0
				end if
			end if
		end if
		break

	case 5
		PlayerObjectCollision(C_TOUCH, -64, -160, 64, 0)
		if CheckResult == true
			if Player.State != Player_State_Death
				if Player.State != Player_State_Drown
					Player.State = Player_State_WaterCurrent
					Player.Animation = ANI_FANROTATE

					Player.ControlMode = -1
					Player.Left = false
					Player.Right = false
					Player.Up = false
					Player.Down = false

					if Player.XPos < Object.XPos
						Player.XVelocity = 0x40000
						Player.Speed = 0x40000
					else
						Player.XVelocity = 0
						Player.Speed = 0
					end if
					Player.YVelocity = 0x40000
				end if
			end if
		else
			if Player.YPos > Object.YPos
				if Player.State == Player_State_WaterCurrent

#platform: Use_Origins
				Player.State = Player_State_Air_NoDropDash
#endplatform

#platform: Use_Standalone
				Player.State = Player_State_Air
#endplatform
					Player.Angle = 0
					Player.ControlMode = 0
					Player.Animation = ANI_WALKING
				end if
			end if
		end if
		break
	end switch
end sub
