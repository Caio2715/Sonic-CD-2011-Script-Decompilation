// Aliases
#alias Object.Value0	:	Object.Timer
#alias Object.Value1	:	Object.YVelocity

#alias 0	:	KUMOKUMO_AIRBORNE

// Different Ground States for the different conditions a Kumo Kumo can come in
#alias 1	:	KUMOKUMO_GOOD_GROUND
#alias 2	:	KUMOKUMO_BAD_GROUND

// State entered to start shooting a Kumo Kumo Web (aka KWeb)
#alias 3	:	KUMOKUMO_SHOOT

// Badnik States (Property Values)
#alias 0	:	PROPVAL_QUALITY_GOOD
#alias 1	:	PROPVAL_QUALITY_BAD

// Kumo Kumo Web Aliases
#alias Object.Value1	:	Object.XVelocity

// SFX Aliases
#alias 2	:	SFX_SHOT



sub ObjectMain
	
	switch Object.State
	case 0
		TempValue0 = 0x4000
		TempValue0 *= 4
		Object.YVelocity += TempValue0
		TempValue0 = Object.YVelocity
		TempValue0 *= 4
		Object.YPos += TempValue0
		if Object.YVelocity > 0
			ObjectTileCollision(0, 0, 24, 0)
			if CheckResult == 1
				Object.YPos += 0x80000
				Object.Frame = 0
				Object.DrawOrder = 3
				if Object.PropertyValue == 0
					Object.State = 1
				else
					Object.State = 2
					Object.Direction=FACING_LEFT
				end if
			end if
		end if
		break
		
	case 1
		if Object.Timer < 45
			Object.Timer++
			Object.Frame++
			Object.Frame &= 3
		else
			Object.YPos -= 0x80000
			Object.Timer = 0
			Object.YVelocity = -0x80000
			Object.YPos += Object.YVelocity
			Object.YVelocity += 0x4000
			Object.State = 0
		end if
		break
		
	case 2
		if Object.Timer < 45
			Object.Timer++
			Object.Frame++
			Object.Frame &= 7
		else
			Object.YPos -= 0x80000
			Object.Timer = 0
			Object.YVelocity = -0x50000
			Object.YPos += Object.YVelocity
			Object.State = 0
		end if
		break
		
	case 3
		if Object.Timer < 16
			Object.Timer++
		else
			CreateTempObject(TypeName[KumoKumoWeb],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].DrawOrder = 4
			if Object.Direction == FACING_RIGHT
				Object[TempObjectPos].XPos += 0x80000
				Object[TempObjectPos].YVelocity = 0x30000
			else
				Object[TempObjectPos].XPos -= 0x80000
				Object[TempObjectPos].YVelocity = -0x30000
			end if
			Object.YPos -= 0x80000
			Object.Timer = 0
			Object.YVelocity = -0x80000
			Object.YPos += Object.YVelocity
			Object.YVelocity += 0x4000
			Object.State = 0
			Object.DrawOrder = 4
			PlayStageSfx(2,0)
		end if
		break
		
	end switch
	
	CallFunction(StageSetup_CheckGoodFuture)
	
end sub


sub ObjectPlayerInteraction
	PlayerObjectCollision(C_TOUCH,-24,-16,24,16)
	if CheckResult == true
		CallFunction(PlayerObject_BadnikBreak)
	end if
	if Object.State == 1
		if Object.Direction == FACING_RIGHT
			PlayerObjectCollision(C_TOUCH,-64,-32,0,16)
			if CheckResult == true
				Object.Direction=FACING_LEFT
				Object.State = 3
				Object.Frame = 0
				Object.Timer = 0
			end if
		else
			PlayerObjectCollision(C_TOUCH,0,-32,64,16)
			if CheckResult == true
				Object.Direction=FACING_RIGHT
				Object.State = 3
				Object.Frame = 0
				Object.Timer = 0
			end if
		end if
	end if
end sub


sub ObjectDraw
	
	switch Object.State
	case 0
		if Object.PropertyValue == 0
			DrawSpriteFX(2, FX_FLIP, Object.XPos, Object.YPos)
		else
			DrawSpriteFX(5, FX_FLIP, Object.XPos, Object.YPos)
		end if
		break
		
	case 1
		TempValue0=Object.Frame
		TempValue0 >>= 1
		DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		break
		
	case 2
		TempValue0=Object.Frame
		TempValue0 >>= 2
		TempValue0 += 3
		DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		break
		
	case 3
		DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
		break
		
	end switch
	
end sub
