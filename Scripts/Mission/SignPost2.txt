sub ObjectMain
	switch Object.State
	case 0
		Object.Frame=1
		TempValue0=Object.YPos
		TempValue0 += 6291456
		if Player.YPos<TempValue0
			if Object.Value2>0
				TempValue0=Object.iXPos
				TempValue0 -= Player.iXPos
				CheckGreater(TempValue0,0)
				TempValue1=CheckResult
				CheckLower(TempValue0,Object.Value2)
				CheckResult&=TempValue1
				if CheckResult == 1
					Stage.NewXBoundary1=Object.iXPos
					Stage.NewXBoundary1 -= Object.Value2
					Stage.NewXBoundary2=Object.XPos
					Stage.NewXBoundary2 >>= 16
					Stage.NewXBoundary2 += Screen.CenterX
				end if
			else
				Stage.NewXBoundary1=Object.XPos
				Stage.NewXBoundary1 >>= 16
				Stage.NewXBoundary1 -= Screen.CenterX
				Stage.NewXBoundary1 -= Screen.CenterX
				Stage.NewXBoundary2=Object.XPos
				Stage.NewXBoundary2 >>= 16
				Stage.NewXBoundary2 += Screen.CenterX
			end if
			if Player.XPos>Object.XPos
				Stage.NewXBoundary1=Object.XPos
				Stage.NewXBoundary1 >>= 16
				Stage.NewXBoundary1 -= Screen.CenterX
				EngineCallback(NOTIFY_TOUCH_SIGNPOST)
				Stage.TimeEnabled=0
				Object.State=1
				Object.Value0=8
				PlaySfx(15,0)
			end if
		end if
		break
	case 1
		Music.Volume--
		Object.Value1++
		if Object.Value1 == 2
			Object.Value1=0
			Object.Frame++
			if Object.Frame == 5
				Rand(TempValue0,48)
				TempValue0 <<= 16
				TempValue0 += Object.XPos
				TempValue0 -= 1572864
				Rand(TempValue1,32)
				TempValue1 <<= 16
				TempValue1 += Object.YPos
				TempValue1 -= 1835008
				CreateTempObject(TypeName[RingSparkle],0,TempValue0,TempValue1)
				Object.Value0--
				if Object.Value0 == 0
					if Object[0].Type == TypeName[BlankObject]
						Object.State=4
						if Object.OutOfBounds == 0
							PlayMusic(1)
						end if
						Object[30].Type=TypeName[ActFinish]
						Object[30].DrawOrder=6
						Object[30].Priority=1
					else
						Player.ControlMode=-1
						Player.Up=0
						Player.Down=0
						Player.Left=0
						Player.JumpPress=0
						Player.JumpHold=0
						Object.State=2
					end if
				end if
			end if
			if Object.Frame == 9
				Rand(TempValue0,48)
				TempValue0 <<= 16
				TempValue0 += Object.XPos
				TempValue0 -= 1572864
				Rand(TempValue1,32)
				TempValue1 <<= 16
				TempValue1 += Object.YPos
				TempValue1 -= 1835008
				CreateTempObject(TypeName[RingSparkle],0,TempValue0,TempValue1)
				Object.Frame=1
			end if
		end if
		TempValue0=40
		TempValue0 += Screen.CenterX
		TempValue0 <<= 16
		TempValue0 += Object.XPos
		if Player.XPos>TempValue0
			Player.XPos=TempValue0
			Player.Speed=0
			Player.XVelocity=0
		end if
		break
	case 2
		if Object[0].Type == TypeName[BlankObject]
			Object.State=4
			if Object.OutOfBounds == 0
				PlayMusic(1)
			end if
			Object[30].Type=TypeName[ActFinish]
			Object[30].DrawOrder=6
			Object[30].Priority=1
		end if
		if Object.PropertyValue>127
			if Player.ControlMode == -1
				Player.Right=1
			end if
		end if
		TempValue0=Screen.CenterX
		TempValue0 <<= 16
		TempValue0 += Object.XPos
		CheckGreater(Player.XPos,TempValue0)
		TempValue1=CheckResult
		CheckLower(Object.PropertyValue,128)
		CheckResult|=TempValue1
		if CheckResult == 1
			if Object.OutOfBounds == 0
				PlayMusic(1)
			end if
			Object.State=3
			Object[30].Type=TypeName[ActFinish]
			Object[30].DrawOrder=6
			Object[30].Priority=1
		end if
		break
	case 3
		if Object.PropertyValue>127
			if Player.ControlMode == -1
				Player.Right=1
			end if
		end if
		TempValue0=40
		TempValue0 += Screen.CenterX
		TempValue0 <<= 16
		TempValue0 += Object.XPos
		if Player.XPos>TempValue0
			Player.XPos=TempValue0
			Player.Speed=0
			Player.XVelocity=0
		end if
		break
	end switch
	if Object.State>1
		if Object.PropertyValue<128
			CheckEqual(Player.State,12)
			TempValue0=CheckResult
			CheckEqual(Player.OutOfBounds,1)
			TempValue0|=CheckResult
			TempValue1=Screen.XOffset
			TempValue1 += Screen.XSize
			TempValue1 += 32
			CheckGreater(Player.iXPos,TempValue1)
			TempValue0|=CheckResult
			if TempValue0 == 1
				Player.State=12
				Player.ControlMode=-1
				Player.JumpHold=0
				Player.Right=0
				Player.Left=0
				Player.Up=0
				Player.Down=0
				Player.XVelocity=0
				Player.YVelocity=0
				Player.Speed=0
				Screen.CameraEnabled=0
			end if
		end if
	end if
	if Object.State>0
		Player.Value4=80
	end if
end sub

	SpriteFrame(-16,-44,32,32,1,183)
	SpriteFrame(-4,-44,8,32,25,150)
	SpriteFrame(-16,-44,32,32,1,216)
	SpriteFrame(-24,-44,48,32,34,132)
	SpriteFrame(-16,-44,32,32,1,183)
	SpriteFrame(-4,-44,8,32,25,150)
	SpriteFrame(-16,-44,32,32,1,216)
	ArrayPos0=32
	while ArrayPos0<1056
		if Object[ArrayPos0].Type == TypeName[SignPost2]
			Object[ArrayPos0].Priority=3
			if Object[ArrayPos0].PropertyValue<128
				Object[ArrayPos0].Value2=Object[ArrayPos0].PropertyValue
			else
				Object[ArrayPos0].Value2=256
				Object[ArrayPos0].Value2 -= Object[ArrayPos0].PropertyValue
			end if
			Object[ArrayPos0].Value2 <<= 4
		end if
		ArrayPos0++
	loop
end sub
