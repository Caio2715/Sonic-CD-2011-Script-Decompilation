function Eggman_FollowGround
	// First move Robotnik 32 pixels up and attempt to find ground
	Object.YPos -= 2097152
	ObjectTileGrip(0, 6, 176, 0)
	if CheckResult == false	

		// No ground was found, so move Robotnik 16 pixels down and try again
		Object.YPos += 1048576
		ObjectTileGrip(0, 6, 176, 0)
		if CheckResult == false

			// Still no collision, try once more
			Object.YPos += 1048576
			ObjectTileGrip(0, 6, 176, 0)
			if CheckResult == false

				// Give up - move Eggman but don't try to follow the ground anymore
				Object.YPos += 1048576
			end if
		end if
	end if
end function

sub ObjectPlayerInteraction

	if Object.LaserActive == true
		if Object.State < 12
			PlayerObjectCollision(C_TOUCH, -12, 48, 12, 176)

			if CheckResult == true
				CallFunction(PlayerObject_Hit)
			end if
		else
			if Player.XPos < Object.XPos
				Player.XPos = Object.XPos
				if Player.RCollisionFlag == true
					CallFunction(PlayerObject_Kill)
				end if
			end if
			
			Player.RCollisionFlag = false
			
			if Object.State < 14
				PlayerObjectCollision(C_TOUCH, -12, 48, 12, 176)
			else
				PlayerObjectCollision(C_TOUCH, -64, -256, 16, 176)
			end if

			if CheckResult == true
				CallFunction(PlayerObject_Kill)
			end if
		end if
	end if
	
	// Limit Tails flight when needed

	if Player.YPos < Object.YPos // Is the Player Object higher up than the Robotnik object?
		if Player.XPos < 234881024 // And has the Player Object not yet passed X Position 3584?
			if Player.State == PlayerObject_HandleTailsFlight
				if Player.Timer < 470
					Player.Timer = 470
				end if
			end if
		end if
		
		if Player.XPos > 948961280 // Alternatively, has the Player Object passed X Position 14480?
			if Player.State == PlayerObject_HandleTailsFlight
				if Player.Timer < 470
					Player.Timer = 470
				end if
			end if
		end if
	end if
	
end sub
