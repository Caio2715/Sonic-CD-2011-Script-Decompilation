sub ObjectMain
	if Player.XPos < Object.XPos
		Object.Direction = FACING_LEFT
	else
		Object.Direction = FACING_RIGHT
	end if
	if Object.Value7 < 600
		Object.Value7++
	else
		if Object.Value7 == 600
			Object.Value7 = 601
			Player.Left = 0
			Player.Right = 0
			Player.Up = 0
			Player.Down = 0
			Player.ControlMode = -1
			CreateTempObject(TypeName[FadeMusic], 3, Object.XPos, Object.YPos)
		end if
	end if
	switch Object.State
	case 0
		Object.Value0++
		switch Object.Frame
		case 4
		case 6
			if Object.Value0 == 4
				Object.Value0 = 0
				Object.Frame++
			end if
			break
		case 5
		case 7
			if Object.Value0 == 3
				Object.Value0 = 0
				Object.Frame++
			end if
			break
		end switch
		if Object.Frame > 7
			Object.Frame = 4
		end if
		if Object.Direction == FACING_RIGHT
			if Object.Value1 < 131072
				Object.Value1 += 8192
			end if
			PlayerObjectCollision(C_TOUCH, -32, -360, 4, 360)
			if CheckResult == 1
				if Player.Gravity == 0
					if Player.Direction == Object.Direction
						PlayerObjectCollision(C_TOUCH, -32, -20, 4, 20)
						if CheckResult == 1
							if Player.Animation == 9
								Object.State = 3
								Object.Frame = 9
								Object.Value2 = -262144
								Object.Value5 = 1
								Object.Value1 = -65536
							else
								if Object.Value7 < 600
									Object.Value7 = 601
									Player.Left = 0
									Player.Right = 0
									Player.Up = 0
									Player.Down = 0
									Player.ControlMode = -1
									CreateTempObject(TypeName[FadeMusic], 3, Object.XPos, Object.YPos)
								end if
#platform: Use_Haptics
								HapticEffect(52, 0, 0, 0)
#endplatform
								Object.State = 2
								Object.Frame = 8
								Object.Value0 = 0
								Player.State = PlayerObject_Hugged
								PlayStageSfx(8, 0)
								SetAchievement(1, 100)
							end if
						else
							Object.State = 1
							Object.Frame = 0
							Object.Value0 = 0
							Object.Value1 = 0
						end if
					else
					end if
				end if
			end if
			if Object.XPos < Object.Value4
				Object.XPos += Object.Value1
			else
				Object.Value1 = 0
				Object.XPos = Object.Value4
				Object.State = 4
				Object.Frame = 0
				Object.Value0 = 0
			end if
		else
			if Object.Value1 > -131072
				Object.Value1 -= 8192
			end if
			PlayerObjectCollision(C_TOUCH, -4, -360, 32, 360)
			if CheckResult == 1
				if Player.Gravity == 0
					if Player.Direction == Object.Direction
						PlayerObjectCollision(C_TOUCH, -32, -20, 4, 20)
						if CheckResult == 1
							if Player.Animation == 9
								Object.State = 3
								Object.Frame = 9
								Object.Value2 = -262144
								Object.Value5 = 1
								Object.Value1 = 65536
							else
								if Object.Value7 < 600
									Object.Value7 = 601
									Player.Left = 0
									Player.Right = 0
									Player.Up = 0
									Player.Down = 0
									Player.ControlMode = -1
									CreateTempObject(TypeName[FadeMusic], 3, Object.XPos, Object.YPos)
								end if
								Object.State = 2
								Object.Frame = 8
								Object.Value0 = 0
								Player.State = PlayerObject_Hugged
								PlayStageSfx(8, 0)
								SetAchievement(1, 100)
#platform: Use_Haptics
								HapticEffect(52, 0, 0, 0)
#endplatform
							end if
						else
							Object.State = 1
							Object.Frame = 0
							Object.Value0 = 0
							Object.Value1 = 0
						end if
					else
					end if
				end if
			end if
			if Object.XPos > Object.Value3
				Object.XPos += Object.Value1
			else
				Object.Value1 = 0
				Object.XPos = Object.Value3
				Object.State = 5
				Object.Frame = 0
				Object.Value0 = 0
			end if
		end if
		break
	case 1
		Object.Value0++
		if Object.Value0 == 12
			Object.Value0 = 0
			Object.Frame++
		end if
		if Object.Frame > 3
			Object.Frame = 0
		end if
		PlayerObjectCollision(C_TOUCH, -4, -360, 4, 360)
		if CheckResult == 0
			if Player.Gravity == 0
				Object.State = 0
				Object.Frame = 4
				Object.Value0 = 0
			end if
		else
			if Player.Gravity == 1
				Object.State = 0
				Object.Frame = 4
				Object.Value0 = 0
			end if
		end if
		break
	case 2
		Player.Speed = 0
		Player.XVelocity = 0
		Player.Direction = Object.Direction
		break
	case 3
		if Object.Value2 > 0
			Object.Frame = 10
		end if
		if Object.Value1 > 0
			Object.Value1 -= 2048
		end if
		if Object.Value1 < 0
			Object.Value1 += 2048
		end if
		Object.XPos += Object.Value1
		if Object.Value5 == 0
			if Object.Value0 == 16
				Object.State = 0
				Object.Frame = 4
				Object.Value0 = 0
			else
				Object.Value0++
			end if
		end if
		break
	case 4
		Object.Value0++
		if Object.Value0 == 12
			Object.Value0 = 0
			Object.Frame++
		end if
		if Object.Frame > 3
			Object.Frame = 0
		end if
		if Object.Direction == FACING_LEFT
			Object.State = 0
			Object.Frame = 4
			Object.Value0 = 0
		end if
		break
	case 5
		Object.Value0++
		if Object.Value0 == 12
			Object.Value0 = 0
			Object.Frame++
		end if
		if Object.Frame > 3
			Object.Frame = 0
		end if
		if Object.Direction == FACING_RIGHT
			Object.State = 0
			Object.Frame = 4
			Object.Value0 = 0
		end if
		break
	end switch
	Object.Value6++
	if Object.State == 7
		if Object.Value6 > 15
			CreateTempObject(TypeName[LoveHeart], 0, Object.XPos, Object.YPos)
			Object[TempObjectPos].iYPos -= 6
			Object[TempObjectPos].InkEffect = 2 //INK_ALPHA
			Object[TempObjectPos].Alpha = 255
			Object.Value6 = 0
		end if
	else
		if Object.Value6 > 41
			CreateTempObject(TypeName[LoveHeart], 0, Object.XPos, Object.YPos)
			Object[TempObjectPos].iYPos -= 6
			Object[TempObjectPos].InkEffect = 2 //INK_ALPHA
			Object[TempObjectPos].Alpha = 255
			Object.Value6 = 0
		end if
	end if
	if Object.Value5 == 0
		ObjectTileGrip(0, 0, 19, 0)
	else
		Object.Value2 += 16384
		Object.YPos += Object.Value2
		ObjectTileCollision(0, 0, 19, 0)
		if CheckResult == 1
			Object.Value2 = 0
			Object.Value5 = 0
		end if
	end if
end sub
