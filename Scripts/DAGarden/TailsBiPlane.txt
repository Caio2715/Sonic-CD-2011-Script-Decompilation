#alias 7: TYPE_TAILSBIPLANE


sub ObjectMain
	if Object.Direction == FACING_RIGHT
		Object.XPos += 65536
	else
		Object.XPos -= 65536
	end if
	switch Object.State
	case 0
	case 3
		Object.Frame = 0
		break
	case 1
		Object.YPos += 32768
		Object.Frame = 3
		break
	case 2
		Object.YPos -= 32768
		Object.Frame = 6
		break
	end switch
	if Object.Value0 > 0
		Object.Value0--
	else
		Rand(Object.State, 3)
		Rand(Object.Value0, 90)
	end if
	if Object.OutOfBounds == 1
		Object.Type = TypeName[BlankObject]
	end if
end sub


sub ObjectDraw
	TempValue0 = Object.AnimationSpeed
	TempValue0 >>= 1
	TempValue0 += Object.Frame
	DrawSpriteFX(TempValue0, FX_FLIP, Object.XPos, Object.YPos)
	Object.AnimationSpeed++
	Object.AnimationSpeed %= 6
end sub
